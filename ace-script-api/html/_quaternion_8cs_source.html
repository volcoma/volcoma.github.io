132 // Summary:
133 // Spherically interpolates between a and b by t. The parameter t is not clamped.
134 //
135 //
136 // Parameters:
137 // a:
138 //
139 // b:
140 //
141 // t:
142 //[FreeFunction("QuaternionScripting::SlerpUnclamped", IsThreadSafe = true)]
143 public static Quaternion SlerpUnclamped(Quaternion a, Quaternion b, float t)
144 {
145 Quaternion z = b;
146
147 float cosTheta = Quaternion.Dot(a, b);
148
149 // If cosTheta < 0, the interpolation will take the long way around the sphere.
150 // To fix this, one quat must be negated.
151 if(cosTheta < 0.0f)
152 {
153 z = -b;
154 cosTheta = -cosTheta;
155 }
156
157 // Perform a linear interpolation when cosTheta is close to 1 to avoid side effect of sin(angle) becoming a zero denominator
158 if(cosTheta > 1.0f - Mathf.kEpsilon)
159 {
160 // Linear interpolation
161 Quaternion result;
162 result.x = Mathf.LerpUnclamped(a.x, z.x, t);
163 result.y = Mathf.LerpUnclamped(a.y, z.y, t);
164 result.z = Mathf.LerpUnclamped(a.z, z.z, t);
165 result.w = Mathf.LerpUnclamped(a.w, z.w, t);
166 return result;
167 }
168 else
169 {
170 // Essential Mathematics, page 467
171 float angle = Mathf.Acos(cosTheta);
172 return (Mathf.Sin((1.0f - t) * angle) * a + Mathf.Sin(t * angle) * z) / Mathf.Sin(angle);
173 }
174 }

350 {
351 float num = rotation.x * 2f;
352 float num2 = rotation.y * 2f;
353 float num3 = rotation.z * 2f;
354 float num4 = rotation.x * num;
355 float num5 = rotation.y * num2;

458 {
459 float num = -0.005729578f;
