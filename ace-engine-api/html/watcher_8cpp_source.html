37auto visit_wild_card_path(const fs::path& path,
38 bool recursive,
39 bool visit_empty,
40 const std::function<bool(const fs::path&)>& visitor) -> std::pair<fs::path, std::string>
41{
42 auto path_filter = get_path_filter_pair(path);
43 if(!path_filter.second.empty())
44 {
45 std::string full = (path_filter.first / path_filter.second).string();
46 size_t wildcard_pos = full.find('*');
47 std::string before = full.substr(0, wildcard_pos);
48 std::string after = full.substr(wildcard_pos + 1);
49 fs::directory_iterator end;
50 fs::error_code err;
51 if(visit_empty && fs::is_empty(path_filter.first, err))
52 {
53 visitor(path_filter.first);
54 }
55 else if(fs::exists(path_filter.first, err))
56 {
57 const auto iterate = [&](auto& it)
58 {
59 for(const auto& entry : it)
60 {
61 std::string current = entry.path().string();
62 size_t before_pos = current.find(before);
63 size_t after_pos = current.find(after);
64 if((before_pos != std::string::npos || before.empty()) &&
65 (after_pos != std::string::npos || after.empty()))
66 {
67 if(visitor(entry.path()))
68 {
69 break;
70 }
71 }
72 }
73 };
74
75 if(recursive)
76 {
77 fs::recursive_directory_iterator it(path_filter.first, err);
78 iterate(it);
79 }
80 else
81 {
82 fs::directory_iterator it(path_filter.first, err);
83 iterate(it);

439 fi.last_mod_time = time;
440 fi.status = watcher::entry_status::modified;
441 fi.type = status.type();

444 }

448 fi.type = status.type();

596{
597 watching_ = true;
598 thread_ = std::thread(
599 [this]()
600 {
601 platform::set_thread_name("fs::watcher");
602 // keep watching for modifications every ms milliseconds
603 using namespace std::literals;
604 while(watching_)
605 {
606 clock_t::duration sleep_time = 99999h;
607
608 // iterate through each watcher and check for modification
609 std::map<std::uint64_t, std::shared_ptr<impl>> watchers;
610 {
611 std::unique_lock<std::mutex> lock(mutex_);
612 watchers = watchers_;
613 }
614
615 for(auto& pair : watchers)
616 {
617 auto watcher = pair.second;
618
619 auto now = clock_t::now();
620
621 auto diff = (watcher->last_poll_ + watcher->poll_interval_) - now;
622 if(diff <= clock_t::duration(0))
623 {
624 watcher->watch();
625 watcher->last_poll_ = now;
626
627 sleep_time = std::min(sleep_time, watcher->poll_interval_);
628 }
629 else
630 {
631 sleep_time = std::min(sleep_time, diff);
632 }
633 }
634
635 std::unique_lock<std::mutex> lock(mutex_);
636 cv_.wait_for(lock, sleep_time);
637 }
638 });

641auto watcher::watch_impl(const fs::path& path,

643 bool initial_list,
644 clock_t::duration poll_interval,

646{

649 if(!wd.watching_)
650 {
651 wd.start();
652 }
653
654 // add a new watcher
655 if(list_callback)
656 {
657 std::string filter;
658 fs::path p = path;
659 // try to see if there's a match for the wild card
660 if(path.string().find('*') != std::string::npos)
661 {
662 auto path_filter = visit_wild_card_path(path,
663 recursive,
664 true,
665 [](const fs::path&)
666 {
667 return true;
668 });
669
670 p = path_filter.first;
671 filter = path_filter.second;
672 }
673 else
674 {
675 fs::error_code err;
676 if(!fs::exists(path, err))
677 {
678 log_path(path);
679 return 0;
680 }
681 }
682
683 static std::atomic<std::uint64_t> free_id = {1};
684
685 auto key = free_id++;
686 {
687 // we do it like this because if initial_list is true we don't want
688 // to call a user callback on a locked mutex
689 auto imp =
690 std::make_shared<impl>(p, filter, recursive, initial_list, poll_interval, std::move(list_callback));
691 std::lock_guard<std::mutex> lock(wd.mutex_);
692 wd.watchers_.emplace(key, std::move(imp));
693 }
694 wd.cv_.notify_all();

696 }
