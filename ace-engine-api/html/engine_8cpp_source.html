289auto engine::process() -> bool
290{

293 auto& sim = ctx.get_cached<simulation>();

296 auto& thr = ctx.get_cached<threader>();

298
299 thr.process();
300
301 sim.run_one_frame(true);
302
303 auto dt = sim.get_delta_time();
304
305 input.manager.before_events_update();
306
307 os::event e{};
308 while(os::poll_event(e))
309 {
310 input.manager.on_os_event(e);
311 ev.on_os_event(ctx, e);
312 }
313 input.manager.after_events_update();
314
315 const auto& window = rend.get_main_window();
316
317 bool should_quit = window == nullptr;
318
319 if(should_quit)
320 {
321 ev.set_play_mode(ctx, false);
322 return false;
323 }
324
325 ev.on_frame_begin(ctx, dt);
326
327 tweeny::update(std::chrono::duration_cast<tweeny::duration_t>(dt));
328
329 ev.on_frame_update(ctx, dt);
330
331 ev.on_frame_render(ctx, dt);
332
333 ev.on_frame_end(ctx, dt);
334
335 get_app_profiler()->swap();
