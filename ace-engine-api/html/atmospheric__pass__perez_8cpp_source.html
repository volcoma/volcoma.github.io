16// Performs piecewise linear interpolation of a Color parameter.
17class dynamic_value_controller
18{
19 using value_type = Color;
20 using key_map = std::map<float, value_type>;
21
22public:
23 dynamic_value_controller(const key_map& keymap) : key_map_(keymap)
24 {
25 }
26
27 value_type get_value(float time) const
28 {
29 auto itUpper = key_map_.upper_bound(time + 1e-6f);
30 auto itLower = itUpper;
31 --itLower;
32
33 if(itLower == key_map_.end())
34 {
35 return itUpper->second;
36 }
37
38 if(itUpper == key_map_.end())
39 {
40 return itLower->second;
41 }
42
43 float lowerTime = itLower->first;
44 const auto& lowerVal = itLower->second;
45 float upperTime = itUpper->first;
46 const auto& upperVal = itUpper->second;
47
48 if(lowerTime == upperTime)
49 {
50 return lowerVal;
51 }
52
53 return interpolate(lowerTime, lowerVal, upperTime, upperVal, time);
54 };
55
56private:
57 value_type interpolate(float lowerTime,
58 const value_type& lowerVal,
59 float upperTime,
60 const value_type& upperVal,
61 float time) const
62 {
63 const float tt = (time - lowerTime) / (upperTime - lowerTime);
64 const auto result = bx::lerp(lowerVal, upperVal, tt);
65 return result;
66 };
67
68 const key_map& key_map_;
69};

165void compute_perez_coeff(float _turbidity, float* _outPerezCoeff)
166{
167 const bx::Vec3 turbidity = {_turbidity, _turbidity, _turbidity};
168 for(uint32_t ii = 0; ii < 5; ++ii)
169 {
170 const bx::Vec3 tmp = bx::mad(ABCDE_t[ii], turbidity, ABCDE[ii]);
171 float* out = _outPerezCoeff + 4 * ii;
172 bx::store(out, tmp);
173 out[3] = 0.0f;
174 }
175}

177float hour_of_day(math::vec3 sun_dir)
178{
179 // Define the ground normal vector (assuming flat and horizontal ground)
180 math::vec3 normal(0.0, -1.0, 0.0);
181
182 auto v1 = sun_dir;
183 auto v2 = normal;
184 auto ref = math::vec3(-1.0f, 0.0f, 0.0f);
185
186 float angle = math::orientedAngle(v1, v2, ref); // angle in [-pi, pi]
187 angle = math::mod(angle, 2 * math::pi<float>()); // angle in [0, 2pi]
188 angle = math::degrees(angle);
189 // The hour angle is 0 at 6:00, 90 at 12:00, and 180 at 18:00
190 // Therefore, we can use a simple linear formula to map the hour angle to the hour of day
191 float hour_of_day = angle / 15;
192
193 // Return the hour of day
194 return hour_of_day;
195}
