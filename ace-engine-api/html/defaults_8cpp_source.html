23void focus_camera_on_bounds(entt::handle camera, const math::bsphere& bounds)
24{
25 auto& trans_comp = camera.get<transform_component>();
26 auto& camera_comp = camera.get<camera_component>();
27 const auto& cam = camera_comp.get_camera();
28
29 math::vec3 cen = bounds.position;
30
31 float aspect = cam.get_aspect_ratio();
32 float fov = cam.get_fov();
33 // Get the radius of a sphere circumscribing the bounds
34 float radius = bounds.radius;
35 // Get the horizontal FOV, since it may be the limiting of the two FOVs to properly
36 // encapsulate the objects
37 float horizontalFOV = math::degrees(2.0f * math::atan(math::tan(math::radians(fov) / 2.0f) * aspect));
38 // Use the smaller FOV as it limits what would get cut off by the frustum
39 float mfov = math::min(fov, horizontalFOV);
40 float dist = radius / (math::sin(math::radians(mfov) / 2.0f));
41
42 trans_comp.look_at(cen);
43 trans_comp.set_position_global(cen - dist * trans_comp.get_z_axis_global());
44 camera_comp.set_ortho_size(radius);
45 camera_comp.update(trans_comp.get_transform_global());
46}

478math::bbox defaults::calc_bounds_global(entt::handle entity)
479{
480 const math::vec3 one = {1.0f, 1.0f, 1.0f};
481 math::bbox bounds = math::bbox(-one, one);
482 auto& ent_trans_comp = entity.get<transform_component>();
483 {
484 auto ent_model_comp = entity.try_get<model_component>();
485 if(ent_model_comp)
486 {
487 ent_model_comp->update_world_bounds(ent_trans_comp.get_transform_global());
488 bounds = ent_model_comp->get_world_bounds();
489 }
490 }
491 return bounds;
492};

494math::bsphere defaults::calc_bounds_sphere_global(entt::handle entity)
495{
496 auto box = calc_bounds_global(entity);

499 auto extends = box.get_extents();
