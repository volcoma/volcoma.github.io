104auto is_parent(entt::const_handle potential_parent, entt::const_handle child) -> bool
105{
106 if(!potential_parent)
107 {
108 return false;
109 }
110 // Traverse up the hierarchy from `child`
111 while(true)
112 {
113 // Access the transform component once per entity
114 const auto* transform = child.try_get<transform_component>();
115 if(!transform)
116 {
117 return false; // Reached the root without finding `potential_parent`
118 }
119 auto parent = transform->get_parent();
120 if(!parent)
121 {
122 return false;
123 }
124
125 if(parent == potential_parent)
126 {
127 return true; // Found the parent relationship
128 }
129
130 child = parent; // Move up the hierarchy
131 }
132}
133auto find_root(entt::const_handle e) -> entt::const_handle
134{
135 // Loop to find the root entity
136 while(true)
137 {
138 // Access the `transform_component` once
139 const auto* transform = e.try_get<transform_component>();
140 if(!transform || !transform->get_parent())
141 {
142 break; // If no parent, we are at the root
143 }
144 e = transform->get_parent(); // Move to the parent entity
145 }
146 return e; // Root entity
147}

149auto are_related(entt::const_handle lhs, entt::const_handle rhs) -> bool
150{
151 return find_root(lhs) == find_root(rhs);

237SAVE(const_entity_handle_link)
238{
239 // Saving entity links is a little more complex than just entities
240 // The rule is as follows.
241 // If we are saving as single entity hierarch :
242 // If the entity link is not part of it :
243 // -> if we are saving to prefab, break the link
244 // -> if we are duplicating resolve the link on load with exsisting scene.
245 entity_flags flags = entity_flags::resolve_with_loaded;
246 entt::const_handle to_save = obj.handle;
247
248 auto& save_ctx = get_save_context();
249
250 bool is_saving_single = save_ctx.save_source.valid();
251 if(is_saving_single)
252 {
253 // is the entity a child of the hierarchy that we are saving?
254 bool save_source_is_parent = is_parent(save_ctx.save_source, obj.handle);
255
256 // if it is an external entity
257 if(!save_source_is_parent)
258 {
259 if(save_ctx.to_prefab)
260 {
261 // when saving prefabs, external entities
262 // should not be saved
263 to_save = {};
264 }
265 else
266 {
267 // when saving entities for duplication purpose, external entities
268 // should not be resolved from the existing scene
269 flags = entity_flags::resolve_with_existing;
270 }
271 }
272 }
273
274 try_save(ar, ser20::make_nvp("flags", flags));
275 save_entity(ar, to_save, flags);
276}

368void flatten_hierarchy(entt::const_handle obj, std::vector<entity_data<entt::const_handle>>& entities)
369{
370 auto& trans_comp = obj.get<transform_component>();
371 const auto& children = trans_comp.get_children();
372
373 entity_data<entt::const_handle> data;
374 data.components.entity = obj;
375
376 entities.emplace_back(std::move(data));
377
378 entities.reserve(entities.size() + children.size());
379 for(const auto& child : children)
380 {
381 flatten_hierarchy(child, entities);
382 }

385template<typename Archive>
386void save_to_archive(Archive& ar, entt::const_handle obj)
387{
388 bool pushed = push_save_context();
389
390 bool is_root = obj.all_of<root_component>();
391 if(!is_root)
392 {
393 const_handle_cast(obj).emplace<root_component>();
394 }
395
396 auto& trans_comp = obj.get<transform_component>();
397
398 std::vector<entity_data<entt::const_handle>> entities;
399 flatten_hierarchy(obj, entities);
400
401 try_save(ar, ser20::make_nvp("entities", entities));
402
403 static const std::string version = "1.0.0";
404 try_save(ar, ser20::make_nvp("version", version));
405
406 if(!is_root)
407 {
408 const_handle_cast(obj).erase<root_component>();
409 }
410
411 pop_save_context(pushed);
412}

461template<typename Archive>
462void save_to_archive(Archive& ar, const entt::registry& reg)
463{
464 bool pushed = push_save_context();
465
466 size_t count = 0;
467 reg.view<transform_component, root_component>().each(
468 [&](auto e, auto&& comp1, auto&& comp2)
469 {
470 count++;
471 });
472
473 try_save(ar, ser20::make_nvp("entities_count", count));
474 reg.view<transform_component, root_component>().each(
475 [&](auto e, auto&& comp1, auto&& comp2)
476 {
477 save_to_archive(ar, entt::const_handle(reg, e));

483template<typename Archive>
484void load_from_archive(Archive& ar, entt::registry& reg)
485{
486 reg.clear();
487 size_t count = 0;
488 try_load(ar, ser20::make_nvp("entities_count", count));
489
490 bool pushed = push_load_context(reg);
491
492 for(size_t i = 0; i < count; ++i)
493 {
494 entt::handle e(reg, reg.create());
495 load_from_archive(ar, e);
496 }

498 pop_load_context(pushed);
