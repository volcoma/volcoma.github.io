16auto interpolate(const std::vector<animation_channel::key<T>>& keys, animation_player::seconds_t time) -> T
17{
18 if(keys.empty())
19 {
20 return {}; // Return default value if there are no keys
21 }
22
23 // Do binary search for keyframe
24 int high = (int)keys.size(), low = -1;
25 while(high - low > 1)
26 {
27 int probe = (high + low) / 2;
28 if(keys[probe].time < time)
29 {
30 low = probe;
31 }
32 else
33 {
34 high = probe;
35 }
36 }
37
38 if(low == -1)
39 {
40 // Before first key, return first key
41 return keys.front().value;
42 }
43
44 if(high == (int)keys.size())
45 {
46 // Beyond last key, return last key
47 return keys.back().value;
48 }
49
50 const auto& key1 = keys[low];
51 const auto& key2 = keys[low + 1];
52
53 // Compute the interpolation factor (0.0 to 1.0)
54 float factor = (time.count() - key1.time.count()) / (key2.time.count() - key1.time.count());
55
56 // Perform the interpolation
57 if constexpr(std::is_same_v<T, math::vec3>)
58 {
59 return math::lerp(key1.value, key2.value, factor);
60 }
61 else if constexpr(std::is_same_v<T, math::quat>)
62 {
63 return math::slerp(key1.value, key2.value, factor);
64 }
65
66 return {};
67}

139void blend_poses(const std::vector<animation_pose>& poses,
140 const std::vector<float>& weights,

142{

144 {

179void blend_space_def::compute_blend(const parameters_t& current_params,
180 std::vector<std::pair<asset_handle<animation_clip>, float>>& out_clips) const
181{
182 // Clear the output vector
183 out_clips.clear();
184
185 // For simplicity, we'll handle a 2D blend space with bilinear interpolation
186 if(parameter_count_ != 2)
187 {
188 // Implement support for other dimensions as needed
189 return;
190 }
191
192 // Find the four closest points for bilinear interpolation
193 // This involves finding the rectangle (grid cell) that the current parameters fall into
194
195 // Collect all parameter values along each axis
196 std::set<float> param0_values;
197 std::set<float> param1_values;
198 for(const auto& point : points_)
199 {
200 param0_values.insert(point.parameters[0]);
201 param1_values.insert(point.parameters[1]);
202 }
203
204 // Convert sets to vectors for indexing
205 std::vector<float> param0_vector(param0_values.begin(), param0_values.end());
206 std::vector<float> param1_vector(param1_values.begin(), param1_values.end());
207
208 // Find indices along each axis
209 auto find_index = [](const std::vector<float>& values, float param) -> size_t
210 {
211 for(size_t i = 0; i < values.size() - 1; ++i)
212 {
213 if(param >= values[i] && param <= values[i + 1])
214 {
215 return i;
216 }
217 }
218 return values.size() - 2; // Return last index if beyond range
219 };
220
221 size_t index0 = find_index(param0_vector, current_params[0]);
222 size_t index1 = find_index(param1_vector, current_params[1]);
223
224 // Get the parameter values at the grid corners
225 float p00 = param0_vector[index0];
226 float p01 = param0_vector[index0 + 1];
227 float p10 = param1_vector[index1];
228 float p11 = param1_vector[index1 + 1];
229
230 // Collect the four corner points
231 std::array<const blend_space_point*, 4> corner_points = {nullptr, nullptr, nullptr, nullptr};
232
233 for(const auto& point : points_)
234 {
235 const auto& params = point.parameters;
236 if(params[0] == p00 && params[1] == p10)
237 corner_points[0] = &point; // Bottom-left
238 if(params[0] == p01 && params[1] == p10)
239 corner_points[1] = &point; // Bottom-right
240 if(params[0] == p00 && params[1] == p11)
241 corner_points[2] = &point; // Top-left
242 if(params[0] == p01 && params[1] == p11)
243 corner_points[3] = &point; // Top-right
244 }
245
246 // Ensure all corner points are found
247 for(const auto* cp : corner_points)
248 {
249 if(!cp)
250 return; // Cannot interpolate without all corner points
251 }
252
253 // Compute interpolation factors
254 float tx = (current_params[0] - p00) / (p01 - p00);
255 float ty = (current_params[1] - p10) / (p11 - p10);
256
257 // Compute weights
258 float w_bl = (1 - tx) * (1 - ty); // Bottom-left
259 float w_br = tx * (1 - ty); // Bottom-right
260 float w_tl = (1 - tx) * ty; // Top-left
261 float w_tr = tx * ty; // Top-right
262
263 // Output the clips and their weights
264 out_clips.emplace_back(corner_points[0]->clip, w_bl);
265 out_clips.emplace_back(corner_points[1]->clip, w_br);
266 out_clips.emplace_back(corner_points[2]->clip, w_tl);
267 out_clips.emplace_back(corner_points[3]->clip, w_tr);

421auto animation_player::update_pose(animation_layer& layer) -> bool
422{
423 auto& state = layer.state;
424 auto& pose = layer.pose;
425 auto& parameters = layer.parameters;
426
427 if(state.blend_space)
428 {
429 // Compute blending weights based on current parameters (e.g., speed and direction)
430 state.blend_space->compute_blend(parameters, state.blend_clips);
431
432 // Sample animations and blend poses
433 state.blend_poses.resize(state.blend_clips.size());
434 for(size_t i = 0; i < state.blend_clips.size(); ++i)
435 {
436 const auto& clip_weight_pair = state.blend_clips[i];
437 sample_animation(clip_weight_pair.first.get().get(), state.elapsed, state.blend_poses[i]);
438 }
439
440 // Blend all poses based on their weights
441 pose.nodes.clear();
442 if(!state.blend_poses.empty())
443 {
444 // Initialize with the first pose
445 pose = state.blend_poses[0];
446 float total_weight = state.blend_clips[0].second;
447
448 for(size_t i = 1; i < state.blend_poses.size(); ++i)
449 {
450 blend_poses(pose,
451 state.blend_poses[i],
452 state.blend_clips[i].second / (total_weight + state.blend_clips[i].second),
453 pose);
454 total_weight += state.blend_clips[i].second;
455 }
456 }
457 return true;
458 }
459 else if(state.clip)
460 {
461 sample_animation(state.clip.get().get(), state.elapsed, pose);
