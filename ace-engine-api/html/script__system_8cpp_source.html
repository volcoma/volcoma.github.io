106auto script_system::find_mono(const rtti::context& ctx) -> mono::compiler_paths
107{
108 bool is_deploy_mode = ctx.has<deploy>();
109
110 mono::compiler_paths result;
111
112 if(is_deploy_mode)
113 {
114 auto mono_dir = fs::resolve_protocol("engine:/mono");
115 result.assembly_dir = fs::absolute(mono_dir / "lib").string();
116 result.config_dir = fs::absolute(mono_dir / "etc").string();
117 }
118 else
119 {
120 const auto& names = mono::get_common_library_names();
121 const auto& library_paths = mono::get_common_library_paths();
122 const auto& config_paths = mono::get_common_config_paths();
123
124 for(size_t i = 0; i < library_paths.size(); ++i)
125 {
126 const auto& library_path = library_paths.at(i);
127 const auto& config_path = config_paths.at(i);
128 std::vector<std::string> paths{library_path};
129 auto found_library = fs::find_library(names, paths);
130
131 if(!found_library.empty())
132 {
133 result.assembly_dir = library_path;
134 result.config_dir = config_path;
135
136 break;
137 }
138 }
139 }
140
141 {
142 const auto& names = mono::get_common_executable_names();
143 const auto& paths = mono::get_common_executable_paths();
144
145 result.msc_executable = fs::find_program(names, paths).string();
146 }
147
148 APPLOG_TRACE("MONO_PATHS:");
149 APPLOG_TRACE("Assembly path - {}", result.assembly_dir);
150 APPLOG_TRACE("Config path - {}", result.config_dir);

496void script_system::check_for_recompile(rtti::context& ctx, delta_t dt)
497{
498 time_since_last_check_ += dt;
499
500 if(time_since_last_check_ >= check_interval)
501 {
502 time_since_last_check_ = {};
503
504 bool should_recompile = needs_recompile.exchange(false);
505
506 if(should_recompile)
507 {
508 auto container = []()
509 {
510 std::lock_guard<std::mutex> lock(container_mutex);
511 auto result = std::move(needs_to_recompile);
512 return result;
513 }();
514
515 for(const auto& protocol : container)
516 {
517 create_compilation_job(ctx, protocol)
518 .then(itc::this_thread::get_id(),
519 [this, &ctx, protocol](auto f)
520 {
521 auto& ev = ctx.get_cached<events>();
522
523 if(ev.is_playing)
524 {
525 return;
526 }
527
528 bool result = f.get();
529 if(result)
530 {
531 ev.on_script_recompile(ctx, protocol);
532 }
533 });
534 }
535 }
536 }
