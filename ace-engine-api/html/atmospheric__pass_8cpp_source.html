11float hour_of_day(math::vec3 sun_dir)
12{
13 // Define the ground normal vector (assuming flat and horizontal ground)
14 math::vec3 normal(0.0, -1.0, 0.0);
15
16 auto v1 = sun_dir;
17 auto v2 = normal;
18 auto ref = math::vec3(-1.0f, 0.0f, 0.0f);
19
20 float angle = math::orientedAngle(v1, v2, ref); // angle in [-pi, pi]
21 angle = math::mod(angle, 2 * math::pi<float>()); // angle in [0, 2pi]
22 angle = math::degrees(angle);
23 // The hour angle is 0 at 6:00, 90 at 12:00, and 180 at 18:00
24 // Therefore, we can use a simple linear formula to map the hour angle to the hour of day
25 float hour_of_day = angle / 15;
26
27 // Return the hour of day
28 return hour_of_day;
29}

48// Function to compute Perez coefficients based on turbidity
49void compute_perez_coeff(float turbidity, float* outPerezCoeff)
50{
51 const bx::Vec3 turbidityVec = {turbidity, turbidity, turbidity};
52 for(uint32_t ii = 0; ii < 5; ++ii)
53 {
54 const bx::Vec3 tmp = bx::mad(ABCDE_t[ii], turbidityVec, ABCDE[ii]);
55 float* out = outPerezCoeff + 4 * ii;
56 bx::store(out, tmp);
57 out[3] = 0.0f; // Set alpha to 0
58 }
