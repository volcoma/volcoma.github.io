817TRANSFORM_INLINE auto transform_t<T, Q>::get_rotation_euler_degrees(vec3_t hint) const noexcept ->
818 typename transform_t<T, Q>::vec3_t
819{
820 auto angles = get_rotation_euler_degrees();
821
822 static auto repeat_working = [](float t, float length)
823 {
824 return (t - (floor(t / length) * length));
825 };
826
827 angles.x = repeat_working(angles.x - hint.x + T(180), T(360)) + hint.x - T(180);
828 angles.y = repeat_working(angles.y - hint.y + T(180), T(360)) + hint.y - T(180);
829 angles.z = repeat_working(angles.z - hint.z + T(180), T(360)) + hint.z - T(180);
830
831 return angles;
832}
