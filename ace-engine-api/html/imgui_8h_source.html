20{
21 return 0 | (uint32_t(_r) << 0) | (uint32_t(_g) << 8) | (uint32_t(_b) << 16) | (uint32_t(_a) << 24);

105inline ImTextureID ToId(const gfx::texture& _handle, uint8_t _mip = 0, uint8_t _flags = IMGUI_FLAGS_ALPHA_BLEND)
106{
107 if(_handle.is_valid() && _handle.is_render_target() && gfx::is_origin_bottom_left())
108 {
109 _flags |= IMGUI_FLAGS_FLIP_UV;
110 }
111
112 return ToId(_handle.native_handle(), _mip, _flags);
113}

116inline ImTextureID ToId(const gfx::texture::ptr& _handle, uint8_t _mip = 0, uint8_t _flags = IMGUI_FLAGS_ALPHA_BLEND)
