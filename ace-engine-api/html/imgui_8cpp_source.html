64 void renderData(gfx::view_id id, ImDrawData* _drawData)
65 {
66 // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates !=
67 // framebuffer coordinates)
68 int fb_width = (int)(_drawData->DisplaySize.x * _drawData->FramebufferScale.x);
69 int fb_height = (int)(_drawData->DisplaySize.y * _drawData->FramebufferScale.y);
70 if(fb_width <= 0 || fb_height <= 0)
71 return;
72
73 gfx::set_view_mode(id, gfx::view_mode::Sequential);
74
75 const gfx::caps* caps = gfx::get_caps();
76 {
77 float ortho[16];
78 float x = _drawData->DisplayPos.x;
79 float y = _drawData->DisplayPos.y;
80 float width = _drawData->DisplaySize.x;
81 float height = _drawData->DisplaySize.y;
82
83 bx::mtxOrtho(ortho, x, x + width, y + height, y, 0.0f, 1000.0f, 0.0f, caps->homogeneousDepth);
84 gfx::set_view_transform(id, nullptr, ortho);
85 gfx::set_view_rect(id, 0, 0, uint16_t(fb_width), uint16_t(fb_height));
86 }
87
88 // (0,0) unless using multi-viewports
89 const ImVec2 clipPos = _drawData->DisplayPos;
90
91 // (1,1) unless using retina display which are often (2,2)
92 const ImVec2 clipScale = _drawData->FramebufferScale;
93
94 // Render command lists
95 for(int32_t ii = 0, num = _drawData->CmdListsCount; ii < num; ++ii)
96 {
97 gfx::transient_vertex_buffer tvb;
98 gfx::transient_index_buffer tib;
99
100 const ImDrawList* drawList = _drawData->CmdLists[ii];
101 uint32_t numVertices = (uint32_t)drawList->VtxBuffer.size();
102 uint32_t numIndices = (uint32_t)drawList->IdxBuffer.size();
103
104 if(!checkAvailTransientBuffers(numVertices, m_layout, numIndices))
105 {
106 // not enough space in transient buffer just quit drawing the rest...
107 break;
108 }
109
110 gfx::alloc_transient_vertex_buffer(&tvb, numVertices, m_layout);
111 gfx::alloc_transient_index_buffer(&tib, numIndices, sizeof(ImDrawIdx) == 4);
112
113 ImDrawVert* verts = (ImDrawVert*)tvb.data;
114 bx::memCopy(verts, drawList->VtxBuffer.begin(), numVertices * sizeof(ImDrawVert));
115
116 ImDrawIdx* indices = (ImDrawIdx*)tib.data;
117 bx::memCopy(indices, drawList->IdxBuffer.begin(), numIndices * sizeof(ImDrawIdx));
118
119 gfx::encoder* encoder = gfx::begin();
120
121 std::set<uint32_t> converted{};
122
123 for(const ImDrawCmd *cmd = drawList->CmdBuffer.begin(), *cmdEnd = drawList->CmdBuffer.end(); cmd != cmdEnd;
124 ++cmd)
125 {
126 m_drawCalls++;
127 if(cmd->UserCallback)
128 {
129 cmd->UserCallback(drawList, cmd);
130 }
131 else if(0 != cmd->ElemCount)
132 {
133 uint64_t state = 0 | BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A | BGFX_STATE_MSAA;
134
135 uint8_t tex_index = 0;
136 gfx::texture_handle th = m_texture;
137 gfx::program_handle program = m_program;
138
139
140
141 if(0 != cmd->TextureId)
142 {
143 ImGui::ImTexture texture;
144 texture.id = cmd->TextureId;
145
146
147 if(0 != (IMGUI_FLAGS_FLIP_UV & texture.s.flags))
148 {
149 for(uint32_t e = 0; e < cmd->ElemCount; ++e)
150 {
151 auto index = indices[cmd->IdxOffset + e];
152 if(converted.emplace(index).second)
153 {
154 auto v = (ImDrawVert*)(verts + index);
155 v->uv.y = 1.0f - v->uv.y;
156 }
157 }
158 }
159
160 state |= 0 != (IMGUI_FLAGS_ALPHA_BLEND & texture.s.flags)
161 ? BGFX_STATE_BLEND_FUNC(BGFX_STATE_BLEND_SRC_ALPHA, BGFX_STATE_BLEND_INV_SRC_ALPHA)
162 : BGFX_STATE_NONE;
163 th = texture.s.handle;
164 tex_index = texture.s.index;
165 if(0 != texture.s.mip)
166 {
167 const float lodEnabled[4] = {float(texture.s.mip), 1.0f, 0.0f, 0.0f};
168 gfx::set_uniform(u_imageLodEnabled, lodEnabled);
169 program = m_imageProgram;
170 }
171 if(texture.s.phandle.idx != gfx::invalid_handle)
172 {
173 program = texture.s.phandle;
174 }
175 }
176 else
177 {
178 state |= BGFX_STATE_BLEND_FUNC(BGFX_STATE_BLEND_SRC_ALPHA, BGFX_STATE_BLEND_INV_SRC_ALPHA);
179 }
180
181 // Project scissor/clipping rectangles into framebuffer space
182 ImVec4 clipRect;
183 clipRect.x = (cmd->ClipRect.x - clipPos.x) * clipScale.x;
184 clipRect.y = (cmd->ClipRect.y - clipPos.y) * clipScale.y;
185 clipRect.z = (cmd->ClipRect.z - clipPos.x) * clipScale.x;
186 clipRect.w = (cmd->ClipRect.w - clipPos.y) * clipScale.y;
187
188 if(clipRect.x < fb_width && clipRect.y < fb_height && clipRect.z >= 0.0f && clipRect.w >= 0.0f)
189 {
190 const uint16_t xx = uint16_t(bx::max(clipRect.x, 0.0f));
191 const uint16_t yy = uint16_t(bx::max(clipRect.y, 0.0f));
192 encoder->setScissor(xx,
193 yy,
194 uint16_t(bx::min(clipRect.z, 65535.0f) - xx),
195 uint16_t(bx::min(clipRect.w, 65535.0f) - yy));
196
197 encoder->setState(state);
198 encoder->setTexture(tex_index, s_tex, th);
199 encoder->setVertexBuffer(0, &tvb, cmd->VtxOffset, numVertices);
200 encoder->setIndexBuffer(&tib, cmd->IdxOffset, cmd->ElemCount);
201 encoder->submit(id, program);
202 }
203
204
205 }
206 }
207
208 gfx::end(encoder);
209 }

249 io.DeltaTime = 1.0f / 60.0f;
250 io.IniFilename = nullptr;
251
252 io.BackendFlags |= ImGuiBackendFlags_RendererHasVtxOffset;
253 io.BackendFlags |= ImGuiBackendFlags_RendererHasViewports; // We can create multi-viewports on the
254 // Renderer side (optional)
255 io.ConfigFlags |= ImGuiConfigFlags_ViewportsEnable;
256 io.ConfigFlags |= ImGuiConfigFlags_DockingEnable;
257
258 // io.ConfigViewportsNoTaskBarIcon = true;
259 io.ConfigDockingTransparentPayload = true;
260
261 auto type = gfx::get_renderer_type();
262 m_program = gfx::create_program(gfx::create_embedded_shader(s_embeddedShaders, type, "vs_ocornut_imgui"),
263 gfx::create_embedded_shader(s_embeddedShaders, type, "fs_ocornut_imgui"),
264 true);
265
266 u_imageLodEnabled = gfx::create_uniform("u_imageLodEnabled", gfx::uniform_type::Vec4);
267 m_imageProgram = gfx::create_program(gfx::create_embedded_shader(s_embeddedShaders, type, "vs_imgui_image"),
268 gfx::create_embedded_shader(s_embeddedShaders, type, "fs_imgui_image"),
269 true);
270
271 m_layout.begin()
272 .add(gfx::attribute::Position, 2, gfx::attribute_type::Float)
273 .add(gfx::attribute::TexCoord0, 2, gfx::attribute_type::Float)
274 .add(gfx::attribute::Color0, 4, gfx::attribute_type::Uint8, true)
275 .end();

277 s_tex = gfx::create_uniform("s_tex", gfx::uniform_type::Sampler);

279 uint8_t* data{};
280 int32_t width{};
281 int32_t height{};
282 {

309 _fontSize,
310 &config,

331 ranges);

333 sizeof(inter_black_ttf),
334 _fontSize,
335 &config,
336 ranges);
337
338
339 m_font[ImGui::Font::Mono] = io.Fonts->AddFontFromMemoryTTF((void*)s_robotoMonoRegularTtf,

341 _fontSize,
342 &config,
343 ranges);
344
345 config.MergeMode = true;
346 config.DstFont = m_font[ImGui::Font::Regular];
347
348 for(uint32_t ii = 0; ii < BX_COUNTOF(s_fontRangeMerge); ++ii)
349 {
350 const FontRangeMerge& frm = s_fontRangeMerge[ii];

352 io.Fonts->AddFontFromMemoryTTF((void*)frm.data, (int)frm.size, _fontSize, &config, frm.ranges);
353 }
354 }
355
356 io.Fonts->GetTexDataAsRGBA32(&data, &width, &height);
357
358 m_texture = gfx::create_texture_2d((uint16_t)width,
359 (uint16_t)height,
360 false,
361 1,
362 gfx::texture_format::BGRA8,
363 0,
364 gfx::copy(data, width * height * 4));

366 auto renderCallback = [this](ace::render_window* window, ImGuiViewport* viewport, void* args)

368 RenderCallback(window, viewport, args);
369 };

372 {
373 };

375 ImGui_ImplOSPP_Init(window, renderCallback, swapCallback);
