147 std::uint8_t gg = *x++;
148 std::uint8_t bb = *x++;

151 // Skip background
152 // if(0 == (rr | gg | bb | aa))
153 // {
154 // continue;
155 // }
156
157 auto hash_key = static_cast<std::uint32_t>(rr + (gg << 8) + (bb << 16) + (aa << 24));

160 if(mapIter != ids.end())
161 {
162 amount = mapIter->second + 1;
163 }
164
165 // Amount of times this ID (color) has been clicked on in buffer
166 ids[hash_key] = amount;
167 max_amount = max_amount > amount ? max_amount : amount;

265void picking_manager::request_pick(math::vec2 pos, const camera& cam)
266{
267 const auto near_clip = cam.get_near_clip();
268 const auto far_clip = cam.get_far_clip();
269 // const auto& pick_pos = *pick_pos_;
270 const auto& frustum = cam.get_frustum();
271 math::vec3 pick_eye;
272 math::vec3 pick_at;
273 math::vec3 pick_up = cam.y_unit_axis(); //{0.0f, 1.0f, 0.0f};
274
275 if(!cam.viewport_to_world(pos, frustum.planes[math::volume_plane::near_plane], pick_eye, true))
276 return;
277
278 if(!cam.viewport_to_world(pos, frustum.planes[math::volume_plane::far_plane], pick_at, true))
279 return;
280
281 camera pick_camera;
282 pick_camera.set_aspect_ratio(1.0f);
283 pick_camera.set_fov(1.0f);
284 pick_camera.set_near_clip(near_clip);
285 pick_camera.set_far_clip(far_clip);
286 pick_camera.look_at(pick_eye, pick_at, pick_up);
287
288 pick_camera_ = pick_camera;
289
290 reading_ = 0;
291 start_readback_ = true;
292}
