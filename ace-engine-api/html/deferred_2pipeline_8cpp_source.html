158auto update_lod_data(lod_data& data,
159 const std::vector<urange32_t>& lod_limits,
160 std::size_t total_lods,
161 float transition_time,
162 float dt,
163 const asset_handle<mesh>& mesh,
164 const math::transform& world,
165 const camera& cam) -> bool
166{
167 if(!mesh)
168 return false;
169
170 if(total_lods <= 1)
171 return true;
172
173 const auto& viewport = cam.get_viewport_size();
174 auto rect = mesh.get()->calculate_screen_rect(world, cam);
175
176 float percent = math::clamp((float(rect.height()) / float(viewport.height)) * 100.0f, 0.0f, 100.0f);
177
178 std::size_t lod = 0;
179 for(size_t i = 0; i < lod_limits.size(); ++i)
180 {
181 const auto& range = lod_limits[i];
182 if(range.contains(urange32_t::value_type(percent)))
183 {
184 lod = i;
185 }
186 }
187
188 lod = math::clamp<std::size_t>(lod, 0, total_lods - 1);
189 if(data.target_lod_index != lod && data.target_lod_index == data.current_lod_index)
190 data.target_lod_index = static_cast<std::uint32_t>(lod);
191
192 if(data.current_lod_index != data.target_lod_index)
193 data.current_time += dt;
194
195 if(data.current_time >= transition_time)
196 {
197 data.current_lod_index = data.target_lod_index;
198 data.current_time = 0.0f;
199 }
200
201 if(percent < 1.0f)
202 return false;
203
204 return true;
205}

690auto deferred::run_lighting_pass(scene& scn,
691 const camera& camera,
692 gfx::render_view& rview,
693 bool apply_shadows,
694 delta_t dt) -> gfx::frame_buffer::ptr
695{
696 APP_SCOPE_PERF("Lighting Pass");
697
698 const auto& view = camera.get_view();
699 const auto& proj = camera.get_projection();
700 const auto& camera_pos = camera.get_position();
701
702 const auto& viewport_size = camera.get_viewport_size();
703
704 const auto& gbuffer = rview.fbo_get("GBUFFER");
705 const auto& rbuffer = rview.fbo_safe_get("RBUFFER");
706 const auto& lbuffer = rview.fbo_get("LBUFFER");
707
708 const auto buffer_size = lbuffer->get_size();
709
710 gfx::render_pass pass("light_buffer_fill");
711 pass.bind(lbuffer.get());
712 pass.set_view_proj(view, proj);
713 pass.clear(BGFX_CLEAR_COLOR, 0, 0.0f, 0);
714
715 scn.registry->view<transform_component, light_component>().each(
716 [&](auto e, auto&& transform_comp_ref, auto&& light_comp_ref)
717 {
718 const auto& light = light_comp_ref.get_light();
719 const auto& generator = light_comp_ref.get_shadowmap_generator();
720 auto world_transform = transform_comp_ref.get_transform_global();
721 world_transform.reset_scale();
722 const auto& light_position = world_transform.get_position();
723 const auto& light_direction = world_transform.z_unit_axis();
724
725 const auto& bounds = light_comp_ref.get_bounds_precise(light_direction);
726 if(!camera.test_obb(bounds, world_transform))
727 {
728 return;
729 }
730
731 irect32_t rect(0, 0, irect32_t::value_type(buffer_size.width), irect32_t::value_type(buffer_size.height));
732 if(light_comp_ref
733 .compute_projected_sphere_rect(rect, light_position, light_direction, camera_pos, view, proj) == 0)
734 return;
735
736 APP_SCOPE_PERF("Lighting Pass Per Light");
737
738 bool has_shadows = light.casts_shadows && apply_shadows;
739
740 const auto& lprogram = has_shadows ? get_light_program(light) : get_light_program_no_shadows(light);
741
742 lprogram.program->begin();
743
744 if(light.type == light_type::directional)
745 {
746 gfx::set_uniform(lprogram.u_light_direction, light_direction);
747 }
748 if(light.type == light_type::point)
749 {
750 float light_data[4] = {light.point_data.range, light.point_data.exponent_falloff, 0.0f, 0.0f};
751
752 gfx::set_uniform(lprogram.u_light_position, light_position);
753 gfx::set_uniform(lprogram.u_light_data, light_data);
754 }
755
756 if(light.type == light_type::spot)
757 {
758 float light_data[4] = {light.spot_data.get_range(),
759 math::cos(math::radians(light.spot_data.get_inner_angle() * 0.5f)),
760 math::cos(math::radians(light.spot_data.get_outer_angle() * 0.5f)),
761 0.0f};
762
763 gfx::set_uniform(lprogram.u_light_direction, light_direction);
764 gfx::set_uniform(lprogram.u_light_position, light_position);
765 gfx::set_uniform(lprogram.u_light_data, light_data);
766 }
767
768 float light_color_intensity[4] = {light.color.value.r,
769 light.color.value.g,
770 light.color.value.b,
771 light.intensity};
772
773 gfx::set_uniform(lprogram.u_light_color_intensity, light_color_intensity);
774 gfx::set_uniform(lprogram.u_camera_position, camera_pos);
775
776 size_t i = 0;
777 for(; i < gbuffer->get_attachment_count(); ++i)
778 {
779 gfx::set_texture(lprogram.s_tex[i], i, gbuffer->get_texture(i));
780 }
781 gfx::set_texture(lprogram.s_tex[i], i, rbuffer);
782 i++;
783 gfx::set_texture(lprogram.s_tex[i], i, ibl_brdf_lut_.get());
784 i++;
785
786 if(has_shadows)
787 {
788 generator.submit_uniforms(i);
789 }
790 gfx::set_scissor(rect.left, rect.top, rect.width(), rect.height());
791 auto topology = gfx::clip_quad(1.0f);
792 gfx::set_state(topology | BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A | BGFX_STATE_BLEND_ADD);
793 gfx::submit(pass.id, lprogram.program->native_handle());
794 gfx::set_state(BGFX_STATE_DEFAULT);
795
796 lprogram.program->end();
797 });
798
799 gfx::discard();
800
801 return lbuffer;
802}

804void deferred::run_reflection_probe_pass(scene& scn, const camera& camera, gfx::render_view& rview, delta_t dt)
805{
806 APP_SCOPE_PERF("Reflection Probe Pass");
807
808 const auto& view = camera.get_view();
809 const auto& proj = camera.get_projection();
810 const auto& camera_pos = camera.get_position();
811
812 const auto& viewport_size = camera.get_viewport_size();
813 const auto& gbuffer = rview.fbo_get("GBUFFER");
814 const auto& rbuffer = rview.fbo_get("RBUFFER");
815
816 const auto buffer_size = rbuffer->get_size();
817
818 gfx::render_pass pass("refl_buffer_fill");
819 pass.bind(rbuffer.get());
820 pass.set_view_proj(view, proj);
821 pass.clear(BGFX_CLEAR_COLOR, 0, 0.0f, 0);
822 std::vector<entt::entity> sorted_probes;
823
824 // Collect all entities with the relevant components
825 scn.registry->view<transform_component, reflection_probe_component>().each(
826 [&](auto e, auto&& transform_comp_ref, auto&& probe_comp_ref)
827 {
828 sorted_probes.emplace_back(e);
829 });
830
831 // Sort the probes based on the method and max range
832 std::sort(std::begin(sorted_probes),
833 std::end(sorted_probes),
834 [&](const auto& lhs, const auto& rhs)
835 {
836 const auto& lhs_comp = scn.registry->get<reflection_probe_component>(lhs);
837 const auto& lhs_probe = lhs_comp.get_probe();
838
839 const auto& rhs_comp = scn.registry->get<reflection_probe_component>(rhs);
840 const auto& rhs_probe = rhs_comp.get_probe();
841
842 // Environment probes should be last
843 if(lhs_probe.method != rhs_probe.method)
844 {
845 return lhs_probe.method < rhs_probe.method; // Environment method is "greater"
846 }
847
848 // If the reflection methods are the same, compare based on the maximum range
849 return lhs_probe.get_max_range() > rhs_probe.get_max_range(); // Smaller ranges first
850 });
851
852 // Render or process the sorted probes
853 for(const auto& e : sorted_probes)
854 {
855 auto& transform_comp_ref = scn.registry->get<transform_component>(e);
856 auto& probe_comp_ref = scn.registry->get<reflection_probe_component>(e);
857
858 const auto& probe = probe_comp_ref.get_probe();
859 const auto& world_transform = transform_comp_ref.get_transform_global();
860 const auto& probe_position = world_transform.get_position();
861 const auto& probe_scale = world_transform.get_scale();
862
863 irect32_t rect(0, 0, irect32_t::value_type(buffer_size.width), irect32_t::value_type(buffer_size.height));
864 if(probe_comp_ref.compute_projected_sphere_rect(rect, probe_position, probe_scale, camera_pos, view, proj) == 0)
865 {
866 continue;
867 }
868
869 const auto& cubemap = probe_comp_ref.get_cubemap();
870
871 ref_probe_program* ref_probe_program = nullptr;
872 float influence_radius = 0.0f;
873 if(probe.type == probe_type::sphere && sphere_ref_probe_program_.program)
874 {
875 ref_probe_program = &sphere_ref_probe_program_;
876 influence_radius =
877 math::max(probe_scale.x, math::max(probe_scale.y, probe_scale.z)) * probe.sphere_data.range;
878 }
879
880 if(probe.type == probe_type::box && box_ref_probe_program_.program)
881 {
882 math::transform t = world_transform;
883 t.scale(probe.box_data.extents);
884 auto u_inv_world = math::inverse(t).get_matrix();
885 float data2[4] = {probe.box_data.extents.x,
886 probe.box_data.extents.y,
887 probe.box_data.extents.z,
888 probe.box_data.transition_distance};
889
890 ref_probe_program = &box_ref_probe_program_;
891
892 gfx::set_uniform(box_ref_probe_program_.u_inv_world, u_inv_world);
893 gfx::set_uniform(box_ref_probe_program_.u_data2, data2);
894
895 influence_radius = math::length(t.get_scale() + probe.box_data.transition_distance);
896 }
897
898 if(ref_probe_program)
899 {
900 float mips = cubemap ? float(cubemap->info.numMips) : 1.0f;
901 float data0[4] = {
902 probe_position.x,
903 probe_position.y,
904 probe_position.z,
905 influence_radius,
906 };
907
908 float data1[4] = {mips, probe.intensity, 0.0f, 0.0f};
909
910 gfx::set_uniform(ref_probe_program->u_data0, data0);
911 gfx::set_uniform(ref_probe_program->u_data1, data1);
912
913 for(size_t i = 0; i < gbuffer->get_attachment_count(); ++i)
914 {
915 gfx::set_texture(ref_probe_program->s_tex[i], i, gbuffer->get_texture(i));
916 }
917
918 gfx::set_texture(ref_probe_program->s_tex_cube, 5, cubemap);
919
920 gfx::set_scissor(rect.left, rect.top, rect.width(), rect.height());
921 auto topology = gfx::clip_quad(1.0f);
922 gfx::set_state(topology | BGFX_STATE_WRITE_RGB | BGFX_STATE_WRITE_A | BGFX_STATE_BLEND_ALPHA);
923
924 ref_probe_program->program->begin();
925 gfx::submit(pass.id, ref_probe_program->program->native_handle());
926 gfx::set_state(BGFX_STATE_DEFAULT);
927 ref_probe_program->program->end();
928 }
929 }
930
931 gfx::discard();
932}
