35void resolve_includes(const fs::path& filePath, std::set<fs::path>& processedFiles)
36{
37 if(!processedFiles.insert(filePath).second)
38 {
39 return; // Avoid processing the same file multiple times
40 }
41
42 std::ifstream file(filePath);
43 if(!file.is_open())
44 {
45 // std::cerr << "Failed to open file: " << filePath << '\n';
46 return;
47 }
48
49 const std::string includeKeyword = "#include";
50 const std::string bgfxIncludePath = "/path/to/bgfx/include"; // Assuming bgfx include path is known
51
52 std::string line;
53 while(std::getline(file, line))
54 {
55 // Trim leading whitespace
56 line.erase(0, line.find_first_not_of(" \t"));
57
58 // Check for #include directive
59 if(line.compare(0, includeKeyword.length(), includeKeyword) == 0)
60 {
61 // Find the start and end of the include path
62 size_t start = line.find_first_of("\"<") + 1;
63 size_t end = line.find_last_of("\">");
64
65 if(start == std::string::npos || end == std::string::npos || start >= end)
66 {
67 // std::cerr << "Invalid include directive: " << line << '\n';
68 continue;
69 }
70
71 std::string includePath = line.substr(start, end - start);
72 fs::path resolvedPath;
73
74 if(line[start - 1] == '<' && line[end] == '>')
75 {
76 // Resolve system include path (e.g., bgfx_shader.sh)
77 // resolvedPath = fs::path(bgfxIncludePath) / includePath;
78 continue;
79 }
80 else
81 {
82 // Resolve local include path relative to the current file
83 resolvedPath = filePath.parent_path() / includePath;
84 }
85
86 resolvedPath = fs::absolute(resolvedPath);
87 // std::cout << "Resolved include: " << resolvedPath << '\n';
88
89 // Recurse into the included file
90 resolve_includes(resolvedPath, processedFiles);
91 }
92 }
93}
