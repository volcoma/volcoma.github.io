93bool RectanglePacker::addRectangle(uint16_t _width, uint16_t _height, uint16_t& _outX, uint16_t& _outY)
94{
95 int best_height, best_index;
96 int32_t best_width;
97 Node* node;
98 Node* prev;
99 _outX = 0;
100 _outY = 0;
101
102 best_height = INT_MAX;
103 best_index = -1;
104 best_width = INT_MAX;
105 for(uint16_t ii = 0, num = uint16_t(m_skyline.size()); ii < num; ++ii)
106 {
107 int32_t yy = fit(ii, _width, _height);
108 if(yy >= 0)
109 {
110 node = &m_skyline[ii];
111 if(((yy + _height) < best_height) || (((yy + _height) == best_height) && (node->width < best_width)))
112 {
113 best_height = uint16_t(yy) + _height;
114 best_index = ii;
115 best_width = node->width;
116 _outX = node->x;
117 _outY = uint16_t(yy);
118 }
119 }
120 }
121
122 if(best_index == -1)
123 {
124 return false;
125 }
126
127 Node newNode(_outX, _outY + _height, _width);
128 m_skyline.insert(m_skyline.begin() + best_index, newNode);
129
130 for(uint16_t ii = uint16_t(best_index + 1), num = uint16_t(m_skyline.size()); ii < num; ++ii)
131 {
132 node = &m_skyline[ii];
133 prev = &m_skyline[ii - 1];
134 if(node->x < (prev->x + prev->width))
135 {
136 uint16_t shrink = uint16_t(prev->x + prev->width - node->x);
137 node->x += shrink;
138 node->width -= shrink;
139 if(node->width <= 0)
140 {
141 m_skyline.erase(m_skyline.begin() + ii);
142 --ii;
143 --num;
144 }
145 else
146 {
147 break;
148 }
149 }
150 else
151 {
152 break;
153 }
154 }
155
156 merge();
157 m_usedSpace += _width * _height;
158 return true;
159}

182int32_t RectanglePacker::fit(uint32_t _skylineNodeIndex, uint16_t _width, uint16_t _height)
183{
184 int32_t width = _width;
185 int32_t height = _height;
186
187 const Node& baseNode = m_skyline[_skylineNodeIndex];
188
189 int32_t xx = baseNode.x, yy;
190 int32_t widthLeft = width;
191 int32_t ii = _skylineNodeIndex;
192
193 if((xx + width) > (int32_t)(m_width - 1))
194 {
195 return -1;
196 }
197
198 yy = baseNode.y;
199 while(widthLeft > 0)
200 {
201 const Node& node = m_skyline[ii];
202 if(node.y > yy)
203 {
204 yy = node.y;
205 }
206
207 if((yy + height) > (int32_t)(m_height - 1))
208 {
209 return -1;
210 }
211
212 widthLeft -= node.width;
213 ++ii;
214 }
215
216 return yy;
217}

219void RectanglePacker::merge()
220{
221 Node* node;
222 Node* next;
223 uint32_t ii;
224
225 for(ii = 0; ii < m_skyline.size() - 1; ++ii)
226 {
227 node = (Node*)&m_skyline[ii];
228 next = (Node*)&m_skyline[ii + 1];
229 if(node->y == next->y)
230 {
231 node->width += next->width;
232 m_skyline.erase(m_skyline.begin() + ii + 1);
233 --ii;
234 }
235 }

310uint16_t Atlas::addRegion(uint16_t _width,
311 uint16_t _height,
312 const uint8_t* _bitmapBuffer,
313 AtlasRegion::Type _type,
314 uint16_t outline)
315{
316 if(m_regionCount >= m_maxRegionCount)
317 {
318 return UINT16_MAX;
319 }
320
321 uint16_t xx = 0;
322 uint16_t yy = 0;
323 uint32_t idx = 0;
324 while(idx < m_usedLayers)
325 {
326 if(m_layers[idx].faceRegion.getType() == _type &&
327 m_layers[idx].packer.addRectangle(_width + 1, _height + 1, xx, yy))
328 {
329 break;
330 }
331
332 idx++;
333 }
334
335 if(idx >= m_usedLayers)
336 {
337 if((idx + _type) > 24 || m_usedFaces >= 6)
338 {
339 return UINT16_MAX;
340 }
341
342 // for (int ii = 0; ii < _type; ++ii)
343 {
344 int ii = 0;
345 AtlasRegion& region = m_layers[idx + ii].faceRegion;
346 region.x = 0;
347 region.y = 0;
348 region.width = m_textureSize;
349 region.height = m_textureSize;
350 region.setMask(_type, m_usedFaces, ii);
351 }
352
353 m_usedLayers++;
354 m_usedFaces++;
355
356 if(!m_layers[idx].packer.addRectangle(_width + 1, _height + 1, xx, yy))
357 {
358 return UINT16_MAX;
359 }
360 }
361
362 AtlasRegion& region = m_regions[m_regionCount];
363 region.x = xx;
364 region.y = yy;
365 region.width = _width;
366 region.height = _height;
367 region.mask = m_layers[idx].faceRegion.mask;
368
369 updateRegion(region, _bitmapBuffer);
370
371 region.x += outline;
372 region.y += outline;
373 region.width -= (outline * 2);
374 region.height -= (outline * 2);
375
376 return m_regionCount++;
377}

379void Atlas::updateRegion(const AtlasRegion& _region, const uint8_t* _bitmapBuffer)
380{
381 uint32_t size = _region.width * _region.height * 4;
382 if(0 < size)
383 {
384 const bgfx::Memory* mem = bgfx::alloc(size);
385 bx::memSet(mem->data, 0, mem->size);
386 if(_region.getType() == AtlasRegion::TYPE_BGRA8)
387 {
388 const uint8_t* inLineBuffer = _bitmapBuffer;
389 uint8_t* outLineBuffer = m_textureBuffer + _region.getFaceIndex() * (m_textureSize * m_textureSize * 4) +
390 (((_region.y * m_textureSize) + _region.x) * 4);
391
392 for(int yy = 0; yy < _region.height; ++yy)
393 {
394 bx::memCopy(outLineBuffer, inLineBuffer, _region.width * 4);
395 inLineBuffer += _region.width * 4;
396 outLineBuffer += m_textureSize * 4;
397 }
398
399 bx::memCopy(mem->data, _bitmapBuffer, mem->size);
400 }
401 else
402 {
403 uint32_t layer = _region.getComponentIndex();
404 const uint8_t* inLineBuffer = _bitmapBuffer;
405 uint8_t* outLineBuffer = (m_textureBuffer + _region.getFaceIndex() * (m_textureSize * m_textureSize * 4) +
406 (((_region.y * m_textureSize) + _region.x) * 4));
407
408 for(int yy = 0; yy < _region.height; ++yy)
409 {
410 for(int xx = 0; xx < _region.width; ++xx)
411 {
412 outLineBuffer[(xx * 4) + layer] = inLineBuffer[xx];
413 }
414
415 bx::memCopy(mem->data + yy * _region.width * 4, outLineBuffer, _region.width * 4);
416 inLineBuffer += _region.width;
417 outLineBuffer += m_textureSize * 4;
418 }
419 }
420
421 bgfx::updateTextureCube(m_textureHandle,
422 0,
423 (uint8_t)_region.getFaceIndex(),
424 0,
425 _region.x,
426 _region.y,
427 _region.width,
428 _region.height,
429 mem);
430 }
431}

453void Atlas::packUV(const AtlasRegion& _region, uint8_t* _vertexBuffer, uint32_t _offset, uint32_t _stride) const
454{
455 int16_t x0 = (int16_t)(((float)_region.x * m_texelSize) - float(INT16_MAX));
456 int16_t y0 = (int16_t)(((float)_region.y * m_texelSize) - float(INT16_MAX));
457 int16_t x1 = (int16_t)((((float)_region.x + _region.width) * m_texelSize) - float(INT16_MAX));
458 int16_t y1 = (int16_t)((((float)_region.y + _region.height) * m_texelSize) - float(INT16_MAX));
459 int16_t ww = (int16_t)((float(INT16_MAX) / 4.0f) * (float)_region.getComponentIndex());
460
461 _vertexBuffer += _offset;
462 switch(_region.getFaceIndex())
463 {
464 case 0: // +X
465 x0 = -x0;
466 x1 = -x1;
467 y0 = -y0;
468 y1 = -y1;
469 writeUV(_vertexBuffer, INT16_MAX, y0, x0, ww);
470 _vertexBuffer += _stride;
471 writeUV(_vertexBuffer, INT16_MAX, y1, x0, ww);
472 _vertexBuffer += _stride;
473 writeUV(_vertexBuffer, INT16_MAX, y1, x1, ww);
474 _vertexBuffer += _stride;
475 writeUV(_vertexBuffer, INT16_MAX, y0, x1, ww);
476 _vertexBuffer += _stride;
477 break;
478
479 case 1: // -X
480 y0 = -y0;
481 y1 = -y1;
482 writeUV(_vertexBuffer, INT16_MIN, y0, x0, ww);
483 _vertexBuffer += _stride;
484 writeUV(_vertexBuffer, INT16_MIN, y1, x0, ww);
485 _vertexBuffer += _stride;
486 writeUV(_vertexBuffer, INT16_MIN, y1, x1, ww);
487 _vertexBuffer += _stride;
488 writeUV(_vertexBuffer, INT16_MIN, y0, x1, ww);
489 _vertexBuffer += _stride;
490 break;
491
492 case 2: // +Y
493 writeUV(_vertexBuffer, x0, INT16_MAX, y0, ww);
494 _vertexBuffer += _stride;
495 writeUV(_vertexBuffer, x0, INT16_MAX, y1, ww);
496 _vertexBuffer += _stride;
497 writeUV(_vertexBuffer, x1, INT16_MAX, y1, ww);
498 _vertexBuffer += _stride;
499 writeUV(_vertexBuffer, x1, INT16_MAX, y0, ww);
500 _vertexBuffer += _stride;
501 break;
502
503 case 3: // -Y
504 y0 = -y0;
505 y1 = -y1;
506 writeUV(_vertexBuffer, x0, INT16_MIN, y0, ww);
507 _vertexBuffer += _stride;
508 writeUV(_vertexBuffer, x0, INT16_MIN, y1, ww);
509 _vertexBuffer += _stride;
510 writeUV(_vertexBuffer, x1, INT16_MIN, y1, ww);
511 _vertexBuffer += _stride;
512 writeUV(_vertexBuffer, x1, INT16_MIN, y0, ww);
513 _vertexBuffer += _stride;
514 break;
515
516 case 4: // +Z
517 y0 = -y0;
518 y1 = -y1;
519 writeUV(_vertexBuffer, x0, y0, INT16_MAX, ww);
520 _vertexBuffer += _stride;
521 writeUV(_vertexBuffer, x0, y1, INT16_MAX, ww);
522 _vertexBuffer += _stride;
523 writeUV(_vertexBuffer, x1, y1, INT16_MAX, ww);
524 _vertexBuffer += _stride;
525 writeUV(_vertexBuffer, x1, y0, INT16_MAX, ww);
526 _vertexBuffer += _stride;
527 break;
528
529 case 5: // -Z
530 x0 = -x0;
531 x1 = -x1;
532 y0 = -y0;
533 y1 = -y1;
534 writeUV(_vertexBuffer, x0, y0, INT16_MIN, ww);
535 _vertexBuffer += _stride;
536 writeUV(_vertexBuffer, x0, y1, INT16_MIN, ww);
537 _vertexBuffer += _stride;
