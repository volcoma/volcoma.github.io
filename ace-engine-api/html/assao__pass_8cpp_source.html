495void assao_pass::create_frame_buffers()
496{
497 m_border = 0;
498
499 m_size[0] = m_width + 2 * m_border;
500 m_size[1] = m_height + 2 * m_border;
501 m_halfSize[0] = (m_size[0] + 1) / 2;
502 m_halfSize[1] = (m_size[1] + 1) / 2;
503 m_quarterSize[0] = (m_halfSize[0] + 1) / 2;
504 m_quarterSize[1] = (m_halfSize[1] + 1) / 2;
505
506 vec4iSet(m_fullResOutScissorRect, m_border, m_border, m_width + m_border, m_height + m_border);
507 vec4iSet(m_halfResOutScissorRect,
508 m_fullResOutScissorRect[0] / 2,
509 m_fullResOutScissorRect[1] / 2,
510 (m_fullResOutScissorRect[2] + 1) / 2,
511 (m_fullResOutScissorRect[3] + 1) / 2);
512
513 int32_t blurEnlarge =
514 cMaxBlurPassCount + bx::max(0, cMaxBlurPassCount - 2); // +1 for max normal blurs, +2 for wide blurs
515 vec4iSet(m_halfResOutScissorRect,
516 bx::max(0, m_halfResOutScissorRect[0] - blurEnlarge),
517 bx::max(0, m_halfResOutScissorRect[1] - blurEnlarge),
518 bx::min(m_halfSize[0], m_halfResOutScissorRect[2] + blurEnlarge),
519 bx::min(m_halfSize[1], m_halfResOutScissorRect[3] + blurEnlarge));
520
521 // Make gbuffer and related textures
522
523 for(int32_t i = 0; i < 4; i++)
524 {
525 m_halfDepths[i] = bgfx::createTexture2D(uint16_t(m_halfSize[0]),
526 uint16_t(m_halfSize[1]),
527 true,
528 1,
529 bgfx::TextureFormat::R16F,
530 BGFX_TEXTURE_COMPUTE_WRITE | SAMPLER_POINT_CLAMP);
531 }
532
533 m_pingPongHalfResultA = bgfx::createTexture2D(uint16_t(m_halfSize[0]),
534 uint16_t(m_halfSize[1]),
535 false,
536 2,
537 bgfx::TextureFormat::RG8,
538 BGFX_TEXTURE_COMPUTE_WRITE);
539 m_pingPongHalfResultB = bgfx::createTexture2D(uint16_t(m_halfSize[0]),
540 uint16_t(m_halfSize[1]),
541 false,
542 2,
543 bgfx::TextureFormat::RG8,
544 BGFX_TEXTURE_COMPUTE_WRITE);
545
546 m_finalResults = bgfx::createTexture2D(uint16_t(m_halfSize[0]),
547 uint16_t(m_halfSize[1]),
548 false,
549 4,
550 bgfx::TextureFormat::RG8,
551 BGFX_TEXTURE_COMPUTE_WRITE | SAMPLER_LINEAR_CLAMP);
552
553 m_normals = bgfx::createTexture2D(uint16_t(m_size[0]),
554 uint16_t(m_size[1]),
555 false,
556 1,
557 bgfx::TextureFormat::RGBA8,
558 BGFX_TEXTURE_COMPUTE_WRITE);
559
560 m_importanceMap = bgfx::createTexture2D(uint16_t(m_quarterSize[0]),
561 uint16_t(m_quarterSize[1]),
562 false,
563 1,

581void assao_pass::destroy_frame_buffers()
582{
583 if(!bgfx::isValid(m_aoMap))
584 {
585 return;
586 }
587
588 for(uint32_t ii = 0; ii < BX_COUNTOF(m_halfDepths); ++ii)
589 {
590 bgfx::destroy(m_halfDepths[ii]);

603void assao_pass::update_uniforms(int32_t _pass, const float* view, const float* proj)
604{
605 vec2Set(m_uniforms.m_viewportPixelSize, 1.0f / (float)m_size[0], 1.0f / (float)m_size[1]);
606 vec2Set(m_uniforms.m_halfViewportPixelSize, 1.0f / (float)m_halfSize[0], 1.0f / (float)m_halfSize[1]);
607
608 vec2Set(m_uniforms.m_viewport2xPixelSize,
609 m_uniforms.m_viewportPixelSize[0] * 2.0f,
610 m_uniforms.m_viewportPixelSize[1] * 2.0f);
611 vec2Set(m_uniforms.m_viewport2xPixelSize_x_025,
612 m_uniforms.m_viewport2xPixelSize[0] * 0.25f,
613 m_uniforms.m_viewport2xPixelSize[1] * 0.25f);
614
615 float depthLinearizeMul =
616 -proj[3 * 4 + 2]; // float depthLinearizeMul = ( clipFar * clipNear ) / ( clipFar - clipNear );
617 float depthLinearizeAdd =
618 proj[2 * 4 + 2]; // float depthLinearizeAdd = clipFar / ( clipFar - clipNear );
619 // correct the handedness issue. need to make sure this below is correct, but I think it is.
620
621 if(depthLinearizeMul * depthLinearizeAdd < 0)
622 {
623 depthLinearizeAdd = -depthLinearizeAdd;
624 }
625
626 vec2Set(m_uniforms.m_depthUnpackConsts, depthLinearizeMul, depthLinearizeAdd);
627
628 float tanHalfFOVY = 1.0f / proj[1 * 4 + 1]; // = tanf( drawContext.Camera.GetYFOV( ) * 0.5f );
629 float tanHalfFOVX = 1.0F / proj[0]; // = tanHalfFOVY * drawContext.Camera.GetAspect( );
630
631 if(bgfx::getRendererType() == bgfx::RendererType::OpenGL)
632 {
633 vec2Set(m_uniforms.m_ndcToViewMul, tanHalfFOVX * 2.0f, tanHalfFOVY * 2.0f);
634 vec2Set(m_uniforms.m_ndcToViewAdd, tanHalfFOVX * -1.0f, tanHalfFOVY * -1.0f);
635 }
636 else
637 {
638 vec2Set(m_uniforms.m_ndcToViewMul, tanHalfFOVX * 2.0f, tanHalfFOVY * -2.0f);
639 vec2Set(m_uniforms.m_ndcToViewAdd, tanHalfFOVX * -1.0f, tanHalfFOVY * 1.0f);
640 }
641
642 m_uniforms.m_effectRadius = bx::clamp(m_settings.m_radius, 0.0f, 100000.0f);
643 m_uniforms.m_effectShadowStrength = bx::clamp(m_settings.m_shadowMultiplier * 4.3f, 0.0f, 10.0f);
644 m_uniforms.m_effectShadowPow = bx::clamp(m_settings.m_shadowPower, 0.0f, 10.0f);
645 m_uniforms.m_effectShadowClamp = bx::clamp(m_settings.m_shadowClamp, 0.0f, 1.0f);
646 m_uniforms.m_effectFadeOutMul = -1.0f / (m_settings.m_fadeOutTo - m_settings.m_fadeOutFrom);
647 m_uniforms.m_effectFadeOutAdd =
648 m_settings.m_fadeOutFrom / (m_settings.m_fadeOutTo - m_settings.m_fadeOutFrom) + 1.0f;
649 m_uniforms.m_effectHorizonAngleThreshold = bx::clamp(m_settings.m_horizonAngleThreshold, 0.0f, 1.0f);
650
651 // 1.2 seems to be around the best trade off - 1.0 means on-screen radius will stop/slow growing when the camera is
652 // at 1.0 distance, so, depending on FOV, basically filling up most of the screen This setting is
653 // viewspace-dependent and not screen size dependent intentionally, so that when you change FOV the effect stays
654 // (relatively) similar.
655 float effectSamplingRadiusNearLimit = (m_settings.m_radius * 1.2f);
656
657 // if the depth precision is switched to 32bit float, this can be set to something closer to 1 (0.9999 is fine)
658 m_uniforms.m_depthPrecisionOffsetMod = 0.9992f;
659
660 // used to get average load per pixel; 9.0 is there to compensate for only doing every 9th InterlockedAdd in
661 // PSPostprocessImportanceMapB for performance reasons
662 m_uniforms.m_loadCounterAvgDiv = 9.0f / (float)(m_quarterSize[0] * m_quarterSize[1] * 255.0);
663
664 // Special settings for lowest quality level - just nerf the effect a tiny bit
665 if(m_settings.m_qualityLevel <= 0)
666 {
667 effectSamplingRadiusNearLimit *= 1.50f;
668
669 if(m_settings.m_qualityLevel < 0)
670 {
671 m_uniforms.m_effectRadius *= 0.8f;
672 }
673 }
674
675 effectSamplingRadiusNearLimit /= tanHalfFOVY; // to keep the effect same regardless of FOV
676
677 m_uniforms.m_effectSamplingRadiusNearLimitRec = 1.0f / effectSamplingRadiusNearLimit;
678
679 m_uniforms.m_adaptiveSampleCountLimit = m_settings.m_adaptiveQualityLimit;
680
681 m_uniforms.m_negRecEffectRadius = -1.0f / m_uniforms.m_effectRadius;
682
683 if(bgfx::getCaps()->originBottomLeft)
684 {
685 vec2Set(m_uniforms.m_perPassFullResCoordOffset, (float)(_pass % 2), 1.0f - (float)(_pass / 2));
686 vec2Set(m_uniforms.m_perPassFullResUVOffset,
687 ((_pass % 2) - 0.0f) / m_size[0],
688 (1.0f - ((_pass / 2) - 0.0f)) / m_size[1]);
689 }
690 else
691 {
692 vec2Set(m_uniforms.m_perPassFullResCoordOffset, (float)(_pass % 2), (float)(_pass / 2));
693 vec2Set(m_uniforms.m_perPassFullResUVOffset,
694 ((_pass % 2) - 0.0f) / m_size[0],
695 ((_pass / 2) - 0.0f) / m_size[1]);
696 }
697
698 m_uniforms.m_invSharpness = bx::clamp(1.0f - m_settings.m_sharpness, 0.0f, 1.0f);
699 m_uniforms.m_passIndex = (float)_pass;
700 vec2Set(m_uniforms.m_quarterResPixelSize, 1.0f / (float)m_quarterSize[0], 1.0f / (float)m_quarterSize[1]);
701
702 float additionalAngleOffset =
703 m_settings.m_temporalSupersamplingAngleOffset; // if using temporal supersampling approach (like "Progressive
704 // Rendering Using Multi-frame Sampling" from GPU Pro 7, etc.)
705 float additionalRadiusScale =
706 m_settings.m_temporalSupersamplingRadiusOffset; // if using temporal supersampling approach (like "Progressive
707 // Rendering Using Multi-frame Sampling" from GPU Pro 7, etc.)
708 const int32_t subPassCount = 5;
709 for(int32_t subPass = 0; subPass < subPassCount; subPass++)
710 {
711 int32_t a = _pass;
712
713 int32_t spmap[5]{0, 1, 4, 3, 2};
714 int32_t b = spmap[subPass];
715
716 float ca, sa;
717 float angle0 = ((float)a + (float)b / (float)subPassCount) * (3.1415926535897932384626433832795f) * 0.5f;
718 angle0 += additionalAngleOffset;
719
720 ca = bx::cos(angle0);
721 sa = bx::sin(angle0);
722
723 float scale = 1.0f + (a - 1.5f + (b - (subPassCount - 1.0f) * 0.5f) / (float)subPassCount) * 0.07f;
724 scale *= additionalRadiusScale;
725
726 vec4Set(m_uniforms.m_patternRotScaleMatrices[subPass], scale * ca, scale * -sa, -scale * sa, -scale * ca);
727 }
728
729 m_uniforms.m_normalsUnpackMul = 2.0f;
730 m_uniforms.m_normalsUnpackAdd = -1.0f;
731
732 m_uniforms.m_detailAOStrength = m_settings.m_detailShadowStrength;
733
734 if(m_settings.m_generateNormals)
735 {
736 bx::mtxIdentity(m_uniforms.m_normalsWorldToViewspaceMatrix);
737 }
738 else
739 {
740 bx::mtxTranspose(m_uniforms.m_normalsWorldToViewspaceMatrix, view);
741 }
742}
