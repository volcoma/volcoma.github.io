37auto extract_lines(hpp::string_view text, int num_lines, int& found_lines) -> hpp::string_view
38{
39 auto pos = hpp::string_view::size_type{0};
40 found_lines = 1;
41 for(int i = 0; i < num_lines; ++i)
42 {
43 pos = text.find('\n', pos);
44 if(pos == hpp::string_view::npos)
45 {
46 break;
47 }
48 ++pos;
49 found_lines = i + 1;
50 }
51 return text.substr(0, pos);
52}

312auto console_log_panel::draw_last_log() -> bool
313{
314 log_entry msg;
315
316
317 {
318 std::lock_guard<std::recursive_mutex> lock(entries_mutex_);
319 for(auto it = std::rbegin(entries_); it != std::rend(entries_); ++it)
320 {
321 const auto& check = *it;
322 if(!enabled_categories_[check.level])
323 {
324 continue;
325 }
326
327 if(!filter_.PassFilter(check.formatted.data(), check.formatted.data() + check.formatted.size()))
328 {
329 continue;
330 }
331 msg = check;
332 break;
333 }
334 }
335
336 if(msg.formatted.empty())
337 {
338 return false;
339 }
340
341 draw_log(msg, 1);
342
343 return true;
344}
