176inline void quote_argument(const std::wstring& argument, std::wstring& command_line, bool force)
177{
178 //
179 // Unless we're told otherwise, don't quote unless we actually
180 // need to do so --- hopefully avoid problems if programs won't
181 // parse quotes properly
182 //
183
184 if(force == false && argument.empty() == false && argument.find_first_of(L" \t\n\v\"") == argument.npos)
185 {
186 command_line.append(argument);
187 }
188 else
189 {
190 command_line.push_back(L'"');
191
192 for(auto it = argument.begin();; ++it)
193 {
194 unsigned number_backslashes = 0;
195
196 while(it != argument.end() && *it == L'\\')
197 {
198 ++it;
199 ++number_backslashes;
200 }
201
202 if(it == argument.end())
203 {
204 //
205 // Escape all backslashes, but let the terminating
206 // double quotation mark we add below be interpreted
207 // as a metacharacter.
208 //
209
210 command_line.append(number_backslashes * 2, L'\\');
211 break;
212 }
213 else if(*it == L'"')
214 {
215 //
216 // Escape all backslashes and the following
217 // double quotation mark.
218 //
219
220 command_line.append(number_backslashes * 2 + 1, L'\\');
221 command_line.push_back(*it);
222 }
223 else
224 {
225 //
226 // Backslashes aren't special here.
227 //
228
229 command_line.append(number_backslashes, L'\\');
230 command_line.push_back(*it);
231 }
232 }
233
234 command_line.push_back(L'"');
235 }
236}

317// env_map_t MapFromWindowsEnvironment()
318// * Imports current Environment in a C-string table
319// * Parses the strings by splitting on the first "=" per line
320// * Creates a map of the variables
321// * Returns the map
322inline env_map_t MapFromWindowsEnvironment()
323{
324 wchar_t* variable_strings_ptr;
325 wchar_t* environment_strings_ptr;
326 std::wstring delimeter(L"=");
327 int del_len = delimeter.length();
328 env_map_t mapped_environment;
329
330 // Get a pointer to the environment block.
331 environment_strings_ptr = GetEnvironmentStringsW();
332 // If the returned pointer is NULL, exit.
333 if(environment_strings_ptr == NULL)
334 {
335 throw OSError("GetEnvironmentStringsW", 0);
336 }
337
338 // Variable strings are separated by NULL byte, and the block is
339 // terminated by a NULL byte.
340
341 variable_strings_ptr = (wchar_t*)environment_strings_ptr;
342
343 // Since the environment map ends with a null, we can loop until we find it.
344 while(*variable_strings_ptr)
345 {
346 // Create a string from Variable String
347 env_string_t current_line(variable_strings_ptr);
348 // Find the first "equals" sign.
349 auto pos = current_line.find(delimeter);
350 // Assuming it's not missing ...
351 if(pos != env_string_t::npos)
352 {
353 // ... parse the key and value.
354 env_string_t key = current_line.substr(0, pos);
355 env_string_t value = current_line.substr(pos + del_len);
356 // Map the entry.
357 mapped_environment[key] = value;
358 }
359 // Jump to next line in the environment map.
360 variable_strings_ptr += std::wcslen(variable_strings_ptr) + 1;
361 }
362 // We're done with the old environment map buffer.
363 FreeEnvironmentStringsW(environment_strings_ptr);
364
365 // Return the map.
366 return mapped_environment;
367}

369// env_vector_t WindowsEnvironmentVectorFromMap(const env_map_t &source_map)
370// * Creates a vector buffer for the new environment string table
371// * Copies in the mapped variables
372// * Returns the vector
373inline env_vector_t WindowsEnvironmentVectorFromMap(const env_map_t& source_map)
374{
375 // Make a new environment map buffer.
376 env_vector_t environment_map_buffer;
377 // Give it some space.
378 environment_map_buffer.reserve(4096);
379
380 // And fill'er up.
381 for(auto kv : source_map)
382 {
383 // Create the line
384 env_string_t current_line(kv.first);
385 current_line += L"=";
386 current_line += kv.second;
387 // Add the line to the buffer.
388 std::copy(current_line.begin(), current_line.end(), std::back_inserter(environment_map_buffer));
389 // Append a null
390 environment_map_buffer.push_back(0);
391 }
392 // Append one last null because of how Windows does it's environment maps.
393 environment_map_buffer.push_back(0);
394
395 return environment_map_buffer;
396}

595static inline int read_all(FILE* fp, std::vector<char>& buf)
596{
597 auto buffer = buf.data();
598 int total_bytes_read = 0;
599 int fill_sz = buf.size();
600
601 while(1)
602 {
603 const int rd_bytes = read_atmost_n(fp, buffer, fill_sz);
604
605 if(rd_bytes == -1)
606 { // Read finished
607 if(total_bytes_read == 0)
608 return -1;
609 break;
610 }
611 else if(rd_bytes == fill_sz)
612 { // Buffer full
613 const auto orig_sz = buf.size();
614 const auto new_sz = orig_sz * 2;
615 buf.resize(new_sz);
616 fill_sz = new_sz - orig_sz;
617
618 // update the buffer pointer
619 buffer = buf.data();
620 total_bytes_read += rd_bytes;
621 buffer += total_bytes_read;
622 }
623 else
624 { // Partial data ? Continue reading
625 total_bytes_read += rd_bytes;
626 fill_sz -= rd_bytes;
627 break;
628 }
629 }
630 buf.erase(buf.begin() + total_bytes_read, buf.end()); // remove extra nulls
631 return total_bytes_read;
632}

1604inline int Popen::poll() noexcept(false)
1605{
1606#ifdef __USING_WINDOWS__
1607 int ret = WaitForSingleObject(process_handle_, 0);
1608 if(ret != WAIT_OBJECT_0)
1609 return -1;
1610

1613 throw OSError("GetExitCodeProcess", 0);
1614

1616 CloseHandle(process_handle_);
1617
1618 return retcode_;
1619#else

1621 return -1; // TODO: ??
1622

1626 int ret = waitpid(child_pid_, &status, WNOHANG);
1627 if(ret == 0)
1628 return -1;
1629
1630 if(ret == child_pid_)
1631 {
1632 if(WIFSIGNALED(status))
1633 {

1635 }

1639 }

1652 // for our process. This child is dead, we cannot get the
1653 // status.
1654 if(errno == ECHILD)
1655 retcode_ = 0;
1656 else
1657 throw OSError("waitpid failed", errno);
1658 }

1683inline void Popen::execute_process() noexcept(false)
1684{
1685#ifdef __USING_WINDOWS__
1686 if(this->shell_)
1687 {
1688 throw OSError("shell not currently supported on windows", 0);
1689 }
1690
1691 void* environment_string_table_ptr = nullptr;
1692 env_vector_t environment_string_vector;
1693 if(this->env_.size())
1694 {
1695 environment_string_vector = util::CreateUpdatedWindowsEnvironmentVector(env_);
1696 environment_string_table_ptr = (void*)environment_string_vector.data();
1697 }
1698
1699 if(exe_name_.length())
1700 {
1701 this->vargs_.insert(this->vargs_.begin(), this->exe_name_);
1702 this->populate_c_argv();
1703 }
1704 this->exe_name_ = vargs_[0];
1705
1706 std::wstring_convert<std::codecvt_utf8_utf16<wchar_t>> converter;
1707 std::wstring argument;
1708 std::wstring command_line;
1709
1710 for(auto arg : this->vargs_)
1711 {
1712 argument = converter.from_bytes(arg);
1713 util::quote_argument(argument, command_line, false);
1714 command_line += L" ";
1715 }
1716
1717 // CreateProcessW can modify szCmdLine so we allocate needed memory
1718 wchar_t* szCmdline = new wchar_t[command_line.size() + 1];
1719 wcscpy_s(szCmdline, command_line.size() + 1, command_line.c_str());
1720 PROCESS_INFORMATION piProcInfo;
1721 STARTUPINFOW siStartInfo;
1722 BOOL bSuccess = FALSE;
1723 DWORD creation_flags = CREATE_UNICODE_ENVIRONMENT | CREATE_NO_WINDOW;
1724
1725 // Set up members of the PROCESS_INFORMATION structure.
1726 ZeroMemory(&piProcInfo, sizeof(PROCESS_INFORMATION));
1727
1728 // Set up members of the STARTUPINFOW structure.
1729 // This structure specifies the STDIN and STDOUT handles for redirection.
1730
1731 ZeroMemory(&siStartInfo, sizeof(STARTUPINFOW));
1732 siStartInfo.cb = sizeof(STARTUPINFOW);
1733
1734 siStartInfo.hStdError = this->stream_.g_hChildStd_ERR_Wr;
1735 siStartInfo.hStdOutput = this->stream_.g_hChildStd_OUT_Wr;
1736 siStartInfo.hStdInput = this->stream_.g_hChildStd_IN_Rd;
1737
1738 siStartInfo.dwFlags |= STARTF_USESTDHANDLES;
1739
1740 // Create the child process.
1741 bSuccess = CreateProcessW(NULL,
1742 szCmdline, // command line
1743 NULL, // process security attributes
1744 NULL, // primary thread security attributes
1745 TRUE, // handles are inherited
1746 creation_flags, // creation flags
1747 environment_string_table_ptr, // use provided environment
1748 NULL, // use parent's current directory
1749 &siStartInfo, // STARTUPINFOW pointer
1750 &piProcInfo); // receives PROCESS_INFORMATION
1751
1752 // If an error occurs, exit the application.
1753 if(!bSuccess)
1754 {
1755 DWORD errorMessageID = ::GetLastError();
1756 throw CalledProcessError("CreateProcess failed: " + util::get_last_error(errorMessageID), errorMessageID);
1757 }
1758
1759 CloseHandle(piProcInfo.hThread);
1760
1761 /*
1762 TODO: use common apis to close linux handles
1763 */
1764
1765 this->process_handle_ = piProcInfo.hProcess;
1766
1767 this->cleanup_future_ = std::async(std::launch::async,
1768 [this]
1769 {
1770 WaitForSingleObject(this->process_handle_, INFINITE);
1771
1772 CloseHandle(this->stream_.g_hChildStd_ERR_Wr);
1773 CloseHandle(this->stream_.g_hChildStd_OUT_Wr);
1774 CloseHandle(this->stream_.g_hChildStd_IN_Rd);
1775 });
1776
1777 /*
1778 NOTE: In the linux version, there is a check to make sure that the process
1779 has been started. Here, we do nothing because CreateProcess will throw
1780 if we fail to create the process.
1781 */
1782
1783#else
1784
1785 int err_rd_pipe, err_wr_pipe;
1786 std::tie(err_rd_pipe, err_wr_pipe) = util::pipe_cloexec();
1787
1788 if(shell_)
1789 {
1790 auto new_cmd = util::join(vargs_);
1791 vargs_.clear();
1792 vargs_.insert(vargs_.begin(), {"/bin/sh", "-c"});
1793 vargs_.push_back(new_cmd);
1794 populate_c_argv();
1795 }
1796
1797 if(exe_name_.length())
1798 {
1799 vargs_.insert(vargs_.begin(), exe_name_);
1800 populate_c_argv();
1801 }
1802 exe_name_ = vargs_[0];
1803
1804 child_pid_ = fork();
1805
1806 if(child_pid_ < 0)
1807 {
1808 close(err_rd_pipe);
1809 close(err_wr_pipe);
1810 throw OSError("fork failed", errno);
1811 }
1812
1813 child_created_ = true;
1814
1815 if(child_pid_ == 0)
1816 {
1817 // Close descriptors belonging to parent
1818 stream_.close_parent_fds();
1819
1820 // Close the read end of the error pipe
1821 close(err_rd_pipe);
1822
1823 detail::Child chld(this, err_wr_pipe);
1824 chld.execute_child();
1825 }
1826 else
1827 {
1828 close(err_wr_pipe); // close child side of pipe, else get stuck in read below
1829
1830 stream_.close_child_fds();
1831
1832 try
1833 {
1834 char err_buf[SP_MAX_ERR_BUF_SIZ] = {
1835 0,
1836 };
1837
1838 int read_bytes = util::read_atmost_n(fdopen(err_rd_pipe, "r"), err_buf, SP_MAX_ERR_BUF_SIZ);
1839 close(err_rd_pipe);
1840
1841 if(read_bytes || strlen(err_buf))
1842 {
1843 // Call waitpid to reap the child process
1844 // waitpid suspends the calling process until the
1845 // child terminates.
1846 int retcode = wait();
1847
1848 // Throw whatever information we have about child failure
1849 throw CalledProcessError(err_buf, retcode);
1850 }
1851 }
1852 catch(std::exception& exp)
1853 {
1854 stream_.cleanup_fds();
1855 throw;
1856 }
1857 }
1858#endif
1859}

1945inline void Child::execute_child()
1946{
1947#ifndef __USING_WINDOWS__
1948 int sys_ret = -1;
1949 auto& stream = parent_->stream_;
1950
1951 try
1952 {
1953 if(stream.write_to_parent_ == 0)
1954 stream.write_to_parent_ = dup(stream.write_to_parent_);
1955
1956 if(stream.err_write_ == 0 || stream.err_write_ == 1)
1957 stream.err_write_ = dup(stream.err_write_);
1958
1959 // Make the child owned descriptors as the
1960 // stdin, stdout and stderr for the child process
1961 auto _dup2_ = [](int fd, int to_fd)
1962 {
1963 if(fd == to_fd)
1964 {
1965 // dup2 syscall does not reset the
1966 // CLOEXEC flag if the descriptors
1967 // provided to it are same.
1968 // But, we need to reset the CLOEXEC
1969 // flag as the provided descriptors
1970 // are now going to be the standard
1971 // input, output and error
1972 util::set_clo_on_exec(fd, false);
1973 }
1974 else if(fd != -1)
1975 {
1976 int res = dup2(fd, to_fd);
1977 if(res == -1)
1978 throw OSError("dup2 failed", errno);
1979 }
1980 };
1981
1982 // Create the standard streams
1983 _dup2_(stream.read_from_parent_, 0); // Input stream
1984 _dup2_(stream.write_to_parent_, 1); // Output stream
1985 _dup2_(stream.err_write_, 2); // Error stream
1986
1987 // Close the duped descriptors
1988 if(stream.read_from_parent_ != -1 && stream.read_from_parent_ > 2)
1989 close(stream.read_from_parent_);
1990
1991 if(stream.write_to_parent_ != -1 && stream.write_to_parent_ > 2)
1992 close(stream.write_to_parent_);
1993
1994 if(stream.err_write_ != -1 && stream.err_write_ > 2)
1995 close(stream.err_write_);
1996
1997 // Close all the inherited fd's except the error write pipe
