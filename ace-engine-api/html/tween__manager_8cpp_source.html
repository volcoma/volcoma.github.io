9tween_id_t tween_manager::start(tween_action tween, const tween_scope_policy& scope_policy)
10{
11 const auto id = tween.get_id();
12
13 if(is_running(tween))
14 {
15 return id;
16 }
17
18 if(is_paused(tween))
19 {
20 resume(tween);
21 return id;
22 }
23
24 auto fill_info = [this, &scope_policy](tween_info& info, tween_action&& tween)
25 {
26 info.tween = std::move(tween);
27 if(scope_policy.scope.empty())
28 {
29 info.scopes.insert(info.scopes.end(), scopes_.begin(), scopes_.end());
30 }
31 else
32 {
33 if(scope_policy.policy == tween_scope_policy::policy_t::independent)
34 {
35 info.scopes.emplace_back(scope_policy.scope);
36 }
37 else if(scope_policy.policy == tween_scope_policy::policy_t::stacked)
38 {
39 info.scopes.insert(info.scopes.end(), scopes_.begin(), scopes_.end());
40 info.scopes.emplace_back(scope_policy.scope);
41 }
42 }
43 };
44
45 auto iter = tweenies_.find(id);
46 if( iter == tweenies_.end() )
47 {
48 auto& info = tweenies_[id];
49 fill_info(info, std::move(tween));
50
51 start_tweeny(info);
52 }
53 else
54 {
55 // Occurs when start yourself from your own callback or start yourself from previous tween process
56 auto& info = pending_tweenies_[id];
57 fill_info(info, std::move(tween));
58 }
59
60 return id;
61}

324void tween_manager::update(duration_t delta)
325{
326 if(delta < duration_t::zero())
327 {
328 delta = duration_t::zero();
329 }
330
331 std::vector<tween_id_t> tweenies_to_start;
332 tweenies_to_start.reserve(pending_tweenies_.size());
333
334 for(auto& kvp : pending_tweenies_)
335 {
336 tweenies_to_start.emplace_back(kvp.first);
337
338 tweenies_[kvp.first] = std::move(kvp.second);
339 }
340 pending_tweenies_.clear();
341
342 for(auto id : tweenies_to_start)
343 {
344 start_tweeny(tweenies_.at(id));
345 }
346
347 for(auto id : get_ids())
348 {
349 auto& tween = tweenies_[id];
350
351 if(tween.depth > 0)
352 {
353 continue;
354 }
355
356 tween.depth++;
357
358 auto state = tween_private::update(tween.tween, delta);
359
360 tween.depth--;
361
362 if(state == state_t::finished)
363 {
364 tweenies_.erase(tween.tween.get_id());
365 }
366 }

408void tween_manager::start_tweeny(tween_info& tween_info)
409{
410 auto& tween = tween_info.tween;
411
412 bool paused_by_scope = false;
413 // First force pause the tweeny before it was started
414 for(const auto& scope : tween_info.scopes)
415 {
416 auto it = std::find_if(paused_scopes_.begin(), paused_scopes_.end(), [&](const auto& sc)
417 {
418 return sc.first == scope;
419 });
420 if(it != paused_scopes_.end())
421 {
422 const auto& scope_pause_key = it->second;
423 tween_private::pause_forced(tween, scope_pause_key);
424 paused_by_scope = true;
425 break;
426 }
427 }
428
429 tween_private::start(tween);
430
431 if(paused_by_scope)
432 {
433 tween_private::pause_forced(tween);
434 }
