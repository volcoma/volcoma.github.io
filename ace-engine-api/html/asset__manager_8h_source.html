341 template<typename T, typename F>
342 auto load_asset_from_file_impl(const std::string& key,
343 load_flags flags,
344 std::recursive_mutex& container_mutex,
345 typename asset_storage<T>::request_container_t& container,
346 F&& load_func) -> asset_handle<T>
347 {
348 if(flags != load_flags::reload)
349 {
350 auto inst = find_asset_impl<T>(key, container_mutex, container);
351 if(inst)
352 {
353 return inst;
354 }
355 }
356
357 std::unique_lock<std::recursive_mutex> lock(container_mutex);
358
359 auto& handle = container[key];
360 // Dispatch the loading
361 if(load_func)
362 {
363 auto uid = add_asset(key);
364 // calling the function on a locked mutex is ok
365 // since we dont expect this to actually
366 // do much except add tasks to the executor
367
368 if(handle.task_id())
369 {
370 pool_.stop(handle.task_id());
371 handle.invalidate();
372 }
373
374 handle.set_internal_ids(uid, key);
375 load_func(pool_, handle, key);
376 }
377
378 return handle;
379 }
