29template<typename T>
30auto make_thumbnail(thumbnail_manager::generator& gen, const asset_handle<T>& asset) -> gfx::texture::ptr
31{
32 auto& thumbnail = gen.thumbnails[asset.uid()];
33 auto current_fbo = thumbnail.get();
34
35 if(gen.remaining > 0 && thumbnail.needs_regeneration)
36 {
37 auto& scn = gen.get_scene();
38 scn.unload();
39 auto& ctx = engine::context();
40 defaults::create_default_3d_scene_for_asset_preview(ctx, scn, asset, {256, 256});
41
42 delta_t dt(0.016667f);
43
44 auto& rpath = ctx.get_cached<rendering_system>();
45 rpath.prepare_scene(scn, dt);
46 auto new_fbo = rpath.render_scene(scn, dt);
47 thumbnail.set(new_fbo);
48 }
49
50 return current_fbo;
51}
