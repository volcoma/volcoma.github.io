67{

123 position += vec3(view[0][1], view[1][1], view[2][1]) * -view[3][1];

140void frustum::recompute_points()
141{
142 // Compute the 8 corner points
143 for(std::size_t i = 0; i < 8; ++i)
144 {
145 const plane& p0 =
146 plane::normalize((i & 1) != 0u ? planes[volume_plane::near_plane] : planes[volume_plane::far_plane]);
147 const plane& p1 = plane::normalize((i & 2) != 0u ? planes[volume_plane::top] : planes[volume_plane::bottom]);
148 const plane& p2 = plane::normalize((i & 4) != 0u ? planes[volume_plane::left] : planes[volume_plane::right]);
149
150 // Compute the point at which the three planes intersect
151 vec3 n0(p0.data);
152 vec3 n1(p1.data);
153 vec3 n2(p2.data);
154
155 vec3 n1_n2 = glm::cross(n1, n2);
156 vec3 n2_n0 = glm::cross(n2, n0);
157 vec3 n0_n1 = glm::cross(n0, n1);
158
159 float cosTheta = glm::dot(n0, n1_n2);
160 float secTheta = 1.0f / cosTheta;
161
162 n1_n2 = n1_n2 * p0.data.w;
163 n2_n0 = n2_n0 * p1.data.w;
164 n0_n1 = n0_n1 * p2.data.w;
165
166 points[i] = -(n1_n2 + n2_n0 + n0_n1) * secTheta;
167
168 } // Next Corner
169}

428auto frustum::swept_sphere_intersect_plane(float& t0,
429 float& t1,
430 const plane& plane,
431 const bsphere& sphere,
432 const vec3& vecSweepDirection) -> bool
433{
434 float b_dot_n = plane::dot_coord(plane, sphere.position);
435 float d_dot_n = plane::dot_normal(plane, vecSweepDirection);
436
437 if(d_dot_n == 0.0f)
438 {
439 if(b_dot_n <= sphere.radius)
440 {
441 // Effectively infinity
442 t0 = 0.0f;
443 t1 = std::numeric_limits<float>::max();
444 return true;
445
446 } // End if infinity
447
448 return false;
449
450 } // End if runs parallel to plane
451
452 // Compute the two possible intersections
453 float tmp0 = (sphere.radius - b_dot_n) / d_dot_n;
454 float tmp1 = (-sphere.radius - b_dot_n) / d_dot_n;
455 t0 = glm::min<float>(tmp0, tmp1);
456 t1 = glm::max<float>(tmp0, tmp1);
457 return true;
458
459 // End if intersection
460}

462auto frustum::test_swept_sphere(const bsphere& sphere, const vec3& vecSweepDirection) const -> bool
463{
464 unsigned int i, nCount = 0;
465 float t0, t1, fDisplacedRadius;
466 float pDisplacements[12];
467 vec3 vDisplacedCenter;
468
469 // Determine all 12 intersection points of the swept sphere with the view
470 // frustum.
471 for(const auto& plane : planes)
472 {
473 // Intersects frustum plane?
474 if(swept_sphere_intersect_plane(t0, t1, plane, sphere, vecSweepDirection))
475 {
476 // TODO: Possibly needs to be < 0?
477 if(t0 >= 0.0f)
478 {
479 pDisplacements[nCount++] = t0;
480 }
481 if(t1 >= 0.0f)
482 {
483 pDisplacements[nCount++] = t1;
484 }
485
486 } // End if intersects
487
488 } // Next plane
489
490 // For all points > 0, displace the sphere along the sweep direction. If the
491 // displaced
492 // sphere falls inside the frustum then we have an intersection.
493 for(i = 0; i < nCount; ++i)
494 {
495 vDisplacedCenter = sphere.position + (vecSweepDirection * pDisplacements[i]);
496 fDisplacedRadius = sphere.radius * 1.1f; // Tolerance.
497 if(test_sphere(bsphere(vDisplacedCenter, fDisplacedRadius)))
498 {
499 return true;
500 }
501
502 } // Next Intersection
503
504 // None of the displaced spheres intersected the frustum
505 return false;
506}

513auto frustum::test_line(const vec3& v1, const vec3& v2) const -> bool
514{
515 int nCode1 = 0, nCode2 = 0;
516 float fDist1, fDist2, t;
517 int nSide1, nSide2;
518 vec3 vDir, vIntersect;
519 unsigned int i;
520
521 // Test each plane
522 for(i = 0; i < 6; ++i)
523 {
524 // Classify each point of the line against the plane.
525 fDist1 = plane::dot_coord(planes[i], v1);
526 fDist2 = plane::dot_coord(planes[i], v2);
527 nSide1 = (fDist1 >= 0) ? 1 : 0;
528 nSide2 = (fDist2 >= 0) ? 1 : 0;
529
530 // Accumulate the classification info to determine
531 // if the edge was spanning any of the planes.
532 nCode1 |= (nSide1 << i);
533 nCode2 |= (nSide2 << i);
534
535 // If the line is completely in front of any plane
536 // then it cannot possibly be intersecting.
537 if(nSide1 == 1 && nSide2 == 1)
538 {
539 return false;
540 }
541
542 // The line is potentially spanning?
543 if((nSide1 ^ nSide2) != 0)
544 {
545 // Compute the point at which the line intersects this plane.
546 vDir = v2 - v1;
547 t = -plane::dot_coord(planes[i], v1) / plane::dot_normal(planes[i], vDir);
548
549 // Truly spanning?
550 if((t >= 0.0f) && (t <= 1.0f))
551 {
552 vIntersect = v1 + (vDir * t);
553 if(test_sphere(bsphere(vIntersect, 0.01f)))
554 {
555 return true;
556 }
557
558 } // End if spanning
559
560 } // End if different sides
561
562 } // Next plane
563
564 // Intersecting?
565 return (nCode1 == 0) || (nCode2 == 0);
566}

568auto frustum::classify_plane(const plane& plane) const -> volume_query
569{
570 unsigned int nInFrontCount = 0;
571 unsigned int nBehindCount = 0;
572
573 // Test frustum points
574 for(unsigned int i = 0; i < 8; ++i)
575 {
576 float fDot = plane::dot_coord(plane, points[i]);
577 if(fDot > 0.0f)
578 {
579 nInFrontCount++;
580 }
581 else if(fDot < 0.0f)
582 {
583 nBehindCount++;
584 }
585
586 } // Next plane
587
588 // frustum entirely in front of plane
589 if(nInFrontCount == 8)
590 {
591 return volume_query::outside;
592 }
593
594 // frustum entire behind plane
595 if(nBehindCount == 8)
596 {
597 return volume_query::inside;
598 }
599
600 // Return intersection (spanning the plane)
601 return volume_query::intersect;
602}

675{

701 {

703 const plane& p2 = frustum.planes[i];
