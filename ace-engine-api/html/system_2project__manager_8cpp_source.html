194void project_manager::save_config()
195{
196 if(has_open_project())
197 {
198 auto& rp = options_.recent_projects;
199 auto project_path = fs::resolve_protocol("app:/");
200 if(std::find_if(std::begin(rp),
201 std::end(rp),
202 [&](const auto& prj)
203 {
204 return project_path.generic_string() == prj.path;
205 }) == std::end(rp))
206 {
207 project prj{project_path.generic_string()};
208 rp.emplace_back(std::move(prj));
209 }
210
211 std::sort(std::begin(rp), std::end(rp), [](const auto& lhs, const auto& rhs)
212 {
213 fs::error_code ec;
214 auto lhs_path = fs::path(lhs.path);
215 auto lhs_time = fs::last_write_time(lhs_path / "settings" / "deploy.cfg", ec);
216
217 auto rhs_path = fs::path(rhs.path);
218 auto rhs_time = fs::last_write_time(rhs_path / "settings" / "deploy.cfg", ec);
219
220 return lhs_time > rhs_time;
221 });
222 }
223
224 fs::error_code err;
225 fs::create_directory(fs::resolve_protocol("editor:/config"), err);
226 const auto project_config_file = fs::resolve_protocol("editor:/config/project.cfg").string();
227 std::ofstream output(project_config_file);
228 auto ar = ser20::create_oarchive_associative(output);
229
230 try_save(ar, ser20::make_nvp("options", options_));
231}
