261void model::recalulate_lod_limits()
262{
263 float upper_limit = 100.0f;
264 lod_limits_.clear();
265 lod_limits_.reserve(mesh_lods_.size());
266
267 float initial = 0.1f;
268 float step = initial / float(mesh_lods_.size());
269 for(size_t i = 0; i < mesh_lods_.size(); ++i)
270 {
271 float lower_limit = 0.0f;
272
273 if(mesh_lods_.size() - 1 != i)
274 {
275 lower_limit = upper_limit * (initial - ((i)*step));
276 }
277
278 lod_limits_.emplace_back(urange32_t::value_type(lower_limit), urange32_t::value_type(upper_limit));
279 upper_limit = lower_limit;
280 }
281}
