171void handle_regular_collision(btPersistentManifold* manifold,
172 const btCollisionObject* obj_a,
173 const btCollisionObject* obj_b,
174 bool enter)
175{
176 int num_contacts = manifold->getNumContacts();
177 if(num_contacts == 0)
178 {
179 return;
180 }
181
182 auto& ctx = ace::engine::context();
183 auto& scripting = ctx.get_cached<ace::script_system>();
184
185 auto a_entity = get_entity_from_user_pointer(obj_a->getUserPointer());
186 auto b_entity = get_entity_from_user_pointer(obj_b->getUserPointer());
187
188 // Log regular collision
189 if(enable_logging)
190 {
191 APPLOG_INFO("Collision {} between entities {} and {}",
192 enter ? "enter" : "exit",
193 get_entity_tag_from_user_pointer(obj_a->getUserPointer()),
194 get_entity_tag_from_user_pointer(obj_b->getUserPointer()));
195 }
196
197 std::vector<ace::manifold_point> contacts;
198 contacts.reserve(num_contacts);
199 for(int i = 0; i < num_contacts; i++)
200 {
201 const btManifoldPoint& contact_point = manifold->getContactPoint(i);
202
203 auto& point = contacts.emplace_back();
204 point.b = from_bullet(contact_point.getPositionWorldOnB());
205 point.a = from_bullet(contact_point.getPositionWorldOnA());
206 point.normal_on_b = from_bullet(contact_point.m_normalWorldOnB);
207 point.normal_on_a = -point.normal_on_b;
208 point.impulse = contact_point.getAppliedImpulse();
209 point.distance = contact_point.getDistance();
210
211 // Access collision details
212 if(enable_logging)
213 {
214 // Log contact points and impulse
215 APPLOG_INFO("Contact Point A: {}", point.a);
216 APPLOG_INFO("Contact Point B: {}", point.b);
217 APPLOG_INFO("Normal on A: {}", point.normal_on_a);
218 APPLOG_INFO("Normal on B: {}", point.normal_on_b);
219 APPLOG_INFO("Impulse: {}", point.impulse);
220 APPLOG_INFO("Distance: {}", point.distance);
221 }
222 }
223
224 if(enter)
225 {
226 scripting.on_collision_enter(a_entity, b_entity, contacts);
227 }
228 else
229 {
230 scripting.on_collision_exit(a_entity, b_entity, contacts);
231 }

234void contact_callback(btPersistentManifold* const& manifold, bool enter)
235{
236 const btCollisionObject* obj_a = manifold->getBody0();
237 const btCollisionObject* obj_b = manifold->getBody1();
238
239 // Determine if either object is a sensor
240 bool is_sensor_a = obj_a->getCollisionFlags() & btCollisionObject::CF_NO_CONTACT_RESPONSE;
241 bool is_sensor_b = obj_b->getCollisionFlags() & btCollisionObject::CF_NO_CONTACT_RESPONSE;
242
243 if(is_sensor_a || is_sensor_b)
244 {
245 // Sensor interaction
246 const btCollisionObject* sensor = is_sensor_a ? obj_a : obj_b;
247 const btCollisionObject* other = is_sensor_a ? obj_b : obj_a;
248
249 // Log or handle sensor "enter/exit" event
250 if(enable_logging)
251 {
252 APPLOG_INFO("Sensor {} {} by {}",
253 enter ? "entered" : "exited",
254 get_entity_tag_from_user_pointer(sensor->getUserPointer()),
255 get_entity_tag_from_user_pointer(other->getUserPointer()));
256 }
257
258 auto& ctx = ace::engine::context();
259 auto& scripting = ctx.get_cached<ace::script_system>();
260
261 auto sensor_entity = get_entity_from_user_pointer(sensor->getUserPointer());
262 auto other_entity = get_entity_from_user_pointer(other->getUserPointer());
263
264 if(enter)
265 {
266 scripting.on_sensor_enter(sensor_entity, other_entity);
267 }
268 else
269 {
270 scripting.on_sensor_exit(sensor_entity, other_entity);
271 }
272 }
273 else
274 {
275 // Regular collision handling
276 handle_regular_collision(manifold, obj_a, obj_b, enter);
277 }

423auto make_rigidbody_shape(physics_component& comp) -> std::shared_ptr<btCompoundShape>
424{
425 auto compound_shapes = comp.get_shapes();
426 if(compound_shapes.empty())
427 {
428 return nullptr;
429 }
430
431 // use an ownning compound shape. When sharing is implemented we can go back to non owning
432 auto cp = std::make_shared<bullet::btCompoundShapeOwning>();
433 for(const auto& s : compound_shapes)
434 {
435 if(hpp::holds_alternative<physics_box_shape>(s.shape))
436 {
437 const auto& shape = hpp::get<physics_box_shape>(s.shape);
438 auto half_extends = shape.extends * 0.5f;
439
440 btBoxShape* box_shape = new btBoxShape({half_extends.x, half_extends.y, half_extends.z});
441
442 btTransform local_transform = btTransform::getIdentity();
443 local_transform.setOrigin(bullet::to_bullet(shape.center));
444 cp->addChildShape(local_transform, box_shape);
445 }
446 else if(hpp::holds_alternative<physics_sphere_shape>(s.shape))
447 {
448 const auto& shape = hpp::get<physics_sphere_shape>(s.shape);
449
450 btSphereShape* sphere_shape = new btSphereShape(shape.radius);
451
452 btTransform local_transform = btTransform::getIdentity();
453 local_transform.setOrigin(bullet::to_bullet(shape.center));
454 cp->addChildShape(local_transform, sphere_shape);
455 }
456 else if(hpp::holds_alternative<physics_capsule_shape>(s.shape))
457 {
458 const auto& shape = hpp::get<physics_capsule_shape>(s.shape);
459
460 btCapsuleShape* capsule_shape = new btCapsuleShape(shape.radius, shape.length);
461
462 btTransform local_transform = btTransform::getIdentity();
463 local_transform.setOrigin(bullet::to_bullet(shape.center));
464 cp->addChildShape(local_transform, capsule_shape);
465 }
466 else if(hpp::holds_alternative<physics_cylinder_shape>(s.shape))
467 {
468 const auto& shape = hpp::get<physics_cylinder_shape>(s.shape);
469
470 btVector3 half_extends(shape.radius, shape.length * 0.5f, shape.radius);
471 btCylinderShape* cylinder_shape = new btCylinderShape(half_extends);
472
473 btTransform local_transform = btTransform::getIdentity();
474 local_transform.setOrigin(bullet::to_bullet(shape.center));
475 cp->addChildShape(local_transform, cylinder_shape);
476 }
477 }
478
479 return cp;
480}

587void recreate_phyisics_body(bullet::world& world, physics_component& comp, bool force = false)
588{
589 bool is_kind_dirty = comp.is_property_dirty(physics_property::kind);
590 bool needs_recreation = force || comp.are_all_properties_dirty();
591
592 auto owner = comp.get_owner();
593
594 if(needs_recreation)
595 {
596 destroy_phyisics_body(world, comp.get_owner(), true);
597 make_rigidbody(world, owner, comp);
598 }
599 else
600 {
601 auto& body = owner.get<bullet::rigidbody>();
602
603 if(comp.is_property_dirty(physics_property::mass) || is_kind_dirty)
604 {
605 update_rigidbody_mass_and_inertia(body, comp);
606 }
607 if(comp.is_property_dirty(physics_property::gravity) || is_kind_dirty)
608 {
609 update_rigidbody_gravity(world, body, comp);
610 }
611 if(comp.is_property_dirty(physics_property::material) || is_kind_dirty)
612 {
613 update_rigidbody_material(body, comp);
614 }
615 if(comp.is_property_dirty(physics_property::sensor) || is_kind_dirty)
616 {
617 update_rigidbody_sensor(body, comp);
618 }
619 if(comp.is_property_dirty(physics_property::shape) || is_kind_dirty)
620 {
621 update_rigidbody_shape(body, comp);
622 update_rigidbody_mass_and_inertia(body, comp);
623 world.dynamics_world->updateSingleAabb(body.internal.get());
624 }
625 if(is_kind_dirty)
626 {
627 update_rigidbody_kind(body, comp);
628 }
629
630 if(!comp.is_kinematic())
631 {
632 if(comp.are_any_properties_dirty())
633 {
634 wake_up(body);
635 }
636 }
637 }
638
639 comp.set_dirty(system_id, false);
640}

642void sync_transforms(bullet::world& world, physics_component& comp, const math::transform& transform)
643{
644 auto owner = comp.get_owner();
645 auto& body = owner.get<bullet::rigidbody>();
646
647 if(!body.internal)
648 {
649 return;
650 }
651
652 const auto& p = transform.get_position();
653 const auto& q = transform.get_rotation();
654 const auto& s = transform.get_scale();
655
656 auto bt_pos = bullet::to_bullet(p);
657 auto bt_rot = bullet::to_bullet(q);
658 btTransform bt_trans(bt_rot, bt_pos);
659 body.internal->setWorldTransform(bt_trans);
660
661 if(body.internal_shape)
662 {
663 auto bt_scale = body.internal_shape->getLocalScaling();
664 auto scale = bullet::from_bullet(bt_scale);
665
666 if(math::any(math::epsilonNotEqual(scale, s, math::epsilon<float>())))
667 {
668 bt_scale = bullet::to_bullet(s);
669 body.internal_shape->setLocalScaling(bt_scale);
670 world.dynamics_world->updateSingleAabb(body.internal.get());
671 }
672 }
673
674 wake_up(body);
675}

725auto add_force(btRigidBody* body, const btVector3& force, force_mode mode) -> bool
726{
727 if(force.fuzzyZero())
728 {
729 return false;
730 }
731 // Apply force based on ForceMode
732 switch(mode)
733 {
734 case force_mode::force: // Continuous force
735 body->applyCentralForce(force);
736 break;
737
738 case force_mode::acceleration:
739 { // Force independent of mass
740 btVector3 acceleration_force = force * body->getMass();
741 body->applyCentralForce(acceleration_force);
742 break;
743 }
744
745 case force_mode::impulse: // Instantaneous impulse
746 body->applyCentralImpulse(force);
747 break;
748
749 case force_mode::velocity_change: // Direct velocity change
750 {
751 btVector3 new_velocity = body->getLinearVelocity() + force; // Accumulate velocity
752 body->setLinearVelocity(new_velocity);
753 break;
754 }
755 }
756 return true;
757}
758
759auto add_torque(btRigidBody* body, const btVector3& torque, force_mode mode) -> bool
760{
761 if(torque.fuzzyZero())
762 {
763 return false;
764 }
765 // Apply force based on ForceMode
766 switch(mode)
767 {
768 case force_mode::force: // Continuous torque
769 body->applyTorque(torque);
770 break;
771
772 case force_mode::acceleration: // Angular acceleration
773 {
774 btVector3 inertia_tensor = body->getInvInertiaDiagLocal();
775 btVector3 angular_acceleration(
776 inertia_tensor.getX() != 0 ? torque.getX() * (1.0f / inertia_tensor.getX()) : 0.0f,
777 inertia_tensor.getY() != 0 ? torque.getY() * (1.0f / inertia_tensor.getY()) : 0.0f,
778 inertia_tensor.getZ() != 0 ? torque.getZ() * (1.0f / inertia_tensor.getZ()) : 0.0f);
779 body->applyTorque(angular_acceleration);
780 }
781 break;
782
783 case force_mode::impulse: // Angular impulse
784 body->applyTorqueImpulse(torque);
785 break;
786
787 case force_mode::velocity_change: // Direct angular velocity change
788 {
789 btVector3 new_velocity = body->getLinearVelocity() + torque; // Accumulate velocity
790 body->setAngularVelocity(new_velocity);
791 break;
792 }
793 }
794
795 return true;
796}

826void bullet_backend::apply_explosion_force(physics_component& comp,
827 float explosion_force,
828 const math::vec3& explosion_position,
829 float explosion_radius,
830 float upwards_modifier,
831 force_mode mode)
832{
833 auto owner = comp.get_owner();
834
835 if(auto bbody = owner.try_get<bullet::rigidbody>())
836 {
837 const auto& body = bbody->internal;
838
839 // Ensure the object is a dynamic rigid body
840 if(body && body->getInvMass() > 0)
841 {
842 // Get the position of the rigid body
843 btVector3 body_position = body->getWorldTransform().getOrigin();
844
845 // Calculate the vector from the explosion position to the body
846 btVector3 direction = body_position - bullet::to_bullet(explosion_position);
847 float distance = direction.length();
848
849 // Skip objects outside the explosion radius
850 if(distance > explosion_radius && explosion_radius > 0.0f)
851 {
852 return;
853 }
854
855 // Normalize the direction vector
856 if(distance > 0.0f)
857 {
858 direction /= distance; // Normalize direction
859 }
860 else
861 {
862 direction.setZero(); // If explosion is at the same position as the body
863 }
864
865 // Apply upwards modifier
866 if(upwards_modifier != 0.0f)
867 {
868 direction.setY(direction.getY() + upwards_modifier);
869 direction.normalize();
870 }
871
872 // Calculate the explosion force magnitude based on distance
873 float attenuation = 1.0f - (distance / explosion_radius);
874 btVector3 force = direction * explosion_force * attenuation;
875
876 if(add_force(body.get(), force, mode))
877 {
878 wake_up(*bbody);
879 }
880 }
881 }
882}
