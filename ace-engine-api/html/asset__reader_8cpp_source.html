57auto validate(const std::string& key, const std::string& compiled_ext, std::string& out) -> bool
58{
59 if(!fs::has_known_protocol(key))
60 {
61 log_unknown_protocol_for_key(key);
62 return false;
63 }
64
65 auto compiled_absolute_path = resolve_compiled_path(key).string() + compiled_ext;
66
67 fs::error_code err;
68 if(!fs::exists(compiled_absolute_path, err))
69 {
70 log_missing_compiled_asset_for_key(compiled_absolute_path);
71
72 compiled_absolute_path = resolve_path(key).string();
73 }
74
75 if(!fs::exists(compiled_absolute_path, err))
76 {
77 log_missing_raw_asset_for_key(key);
78 return false;
79 }
80
81 out = compiled_absolute_path;
82 return true;
83}
