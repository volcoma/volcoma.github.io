16{
17 math::bsphere result;

19 if(light_.type == light_type::point)

21 result = math::bsphere(math::vec3(0.0f, 0.0f, 0.0f), light_.point_data.range);
22 }
23 else if(light_.type == light_type::spot)
24 {
25 float range = light_.spot_data.get_range();
26
27 if(light_direction)
28 {
29 float clamped_inner_cone_angle =
30 math::radians(math::clamp(light_.spot_data.get_inner_angle(), 0.0f, 89.0f));
31 float clamped_outer_cone_angle = math::clamp(math::radians(light_.spot_data.get_outer_angle()),
32 clamped_inner_cone_angle + 0.001f,
33 math::radians(89.0f) + 0.001f);
34 float cos_outer_cone = math::cos(clamped_outer_cone_angle);
35 // Use the law of cosines to find the distance to the furthest edge of the
36 // spotlight cone from a
37 // position that is halfway down the spotlight direction
38 const float radius = math::sqrt(1.25f * range * range - range * range * cos_outer_cone);
39 math::vec3 center = math::vec3(0.0f, 0.0f, 0.0f) + 0.5f * (*light_direction) * range;
40
41 result = math::bsphere(center, radius);
42 }
43 else
44 {
45 result = math::bsphere(math::vec3(0.0f, 0.0f, 0.0f), range);
46 }
47 }
48 else
49 {
50 result = math::bsphere(math::vec3(0.0f, 0.0f, 0.0f), 999999999.0f);
51 }

53 return result;
